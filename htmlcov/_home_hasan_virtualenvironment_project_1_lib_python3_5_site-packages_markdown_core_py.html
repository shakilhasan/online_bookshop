109 def registerExtensions(self, extensions, configs): 

110 """ 

111 Register extensions with this instance of Markdown. 

112 

113 Keyword arguments: 

114 

115 * extensions: A list of extensions, which can either 

116 be strings or objects. 

117 * configs: A dictionary mapping extension names to config options. 

118 

119 """ 

120 for ext in extensions: 

121 if isinstance(ext, str): 

122 ext = self.build_extension(ext, configs.get(ext, {})) 

123 if isinstance(ext, Extension): 

124 ext._extendMarkdown(self) 

125 logger.debug( 

126 'Successfully loaded extension "%s.%s".' 

127 % (ext.__class__.__module__, ext.__class__.__name__) 

128 ) 

129 elif ext is not None: 

130 raise TypeError( 

131 'Extension "{}.{}" must be of type: "{}.{}"'.format( 

132 ext.__class__.__module__, ext.__class__.__name__, 

133 Extension.__module__, Extension.__name__ 

134 ) 

135 ) 

136 return self

138 def build_extension(self, ext_name, configs): 

139 """ 

140 Build extension from a string name, then return an instance. 

141 

142 First attempt to load an entry point. The string name must be registered as an entry point in the 

143 `markdown.extensions` group which points to a subclass of the `markdown.extensions.Extension` class. 

144 If multiple distributions have registered the same name, the first one found is returned. 

145 

146 If no entry point is found, assume dot notation (`path.to.module:ClassName`). Load the specified class and 

147 return an instance. If no class is specified, import the module and call a `makeExtension` function and return 

148 the Extension instance returned by that function. 

149 """ 

150 configs = dict(configs) 

151 

152 entry_points = [ep for ep in util.INSTALLED_EXTENSIONS if ep.name == ext_name] 

153 if entry_points: 

154 ext = entry_points[0].load() 

155 return ext(**configs) 

156 

157 # Get class name (if provided): `path.to.module:ClassName` 

158 ext_name, class_name = ext_name.split(':', 1) if ':' in ext_name else (ext_name, '') 

159 

160 try: 

161 module = importlib.import_module(ext_name) 

162 logger.debug( 

163 'Successfully imported extension module "%s".' % ext_name 

164 ) 

165 except ImportError as e: 

166 message = 'Failed loading extension "%s".' % ext_name 

167 e.args = (message,) + e.args[1:] 

168 raise 

169 

170 if class_name: 

171 # Load given class name from module. 

172 return getattr(module, class_name)(**configs) 

173 else: 

174 # Expect makeExtension() function to return a class. 

175 try: 

176 return module.makeExtension(**configs) 

177 except AttributeError as e: 

178 message = e.args[0] 

179 message = "Failed to initiate extension " \ 

180 "'%s': %s" % (ext_name, message) 

181 e.args = (message,) + e.args[1:] 

182 raise

224 def convert(self, source): 

225 """ 

226 Convert markdown to serialized XHTML or HTML. 

227 

228 Keyword arguments: 

229 

230 * source: Source text as a Unicode string. 

231 

232 Markdown processing takes place in five steps: 

233 

234 1. A bunch of "preprocessors" munge the input text. 

235 2. BlockParser() parses the high-level structural elements of the 

236 pre-processed text into an ElementTree. 

237 3. A bunch of "treeprocessors" are run against the ElementTree. One 

238 such treeprocessor runs InlinePatterns against the ElementTree, 

239 detecting inline markup. 

240 4. Some post-processors are run against the text after the ElementTree 

241 has been serialized into text. 

242 5. The output is written to a string. 

243 

244 """ 

245 

246 # Fixup the source text 

247 if not source.strip(): 

248 return '' # a blank unicode string 

249 

250 try: 

251 source = str(source) 

252 except UnicodeDecodeError as e: # pragma: no cover 

253 # Customise error message while maintaining original trackback 

254 e.reason += '. -- Note: Markdown only accepts unicode input!' 

255 raise 

256 

257 # Split into lines and run the line preprocessors. 

258 self.lines = source.split("\n") 

259 for prep in self.preprocessors: 

260 self.lines = prep.run(self.lines) 

261 

262 # Parse the high-level elements. 

263 root = self.parser.parseDocument(self.lines).getroot() 

264 

265 # Run the tree-processors 

266 for treeprocessor in self.treeprocessors: 

267 newRoot = treeprocessor.run(root) 

268 if newRoot is not None: 

269 root = newRoot 

270 

271 # Serialize _properly_. Strip top-level tags. 

272 output = self.serializer(root) 

273 if self.stripTopLevelTags: 

274 try: 

275 start = output.index( 

276 '<%s>' % self.doc_tag) + len(self.doc_tag) + 2 

277 end = output.rindex('</%s>' % self.doc_tag) 

278 output = output[start:end].strip() 

279 except ValueError: # pragma: no cover 

280 if output.strip().endswith('<%s />' % self.doc_tag): 

281 # We have an empty document 

282 output = '' 

283 else: 

284 # We have a serious problem 

285 raise ValueError('Markdown failed to strip top-level ' 

286 'tags. Document=%r' % output.strip()) 

287 

288 # Run the text post-processors 

289 for pp in self.postprocessors: 

290 output = pp.run(output) 

291 

292 return output.strip()

294 def convertFile(self, input=None, output=None, encoding=None): 

295 """Converts a markdown file and returns the HTML as a unicode string. 

296 

297 Decodes the file using the provided encoding (defaults to utf-8), 

298 passes the file content to markdown, and outputs the html to either 

299 the provided stream or the file with provided name, using the same 

300 encoding as the source file. The 'xmlcharrefreplace' error handler is 

301 used when encoding the output. 

302 

303 **Note:** This is the only place that decoding and encoding of unicode 

304 takes place in Python-Markdown. (All other code is unicode-in / 

305 unicode-out.) 

306 

307 Keyword arguments: 

308 

309 * input: File object or path. Reads from stdin if `None`. 

310 * output: File object or path. Writes to stdout if `None`. 

311 * encoding: Encoding of input and output files. Defaults to utf-8. 

312 

313 """ 

314 

315 encoding = encoding or "utf-8" 

316 

317 # Read the source 

318 if input: 

319 if isinstance(input, str): 

320 input_file = codecs.open(input, mode="r", encoding=encoding) 

321 else: 

322 input_file = codecs.getreader(encoding)(input) 

323 text = input_file.read() 

324 input_file.close() 

325 else: 

326 text = sys.stdin.read() 

327 if not isinstance(text, str): # pragma: no cover 

328 text = text.decode(encoding) 

329 

330 text = text.lstrip('\ufeff') # remove the byte-order mark 

331 

332 # Convert 

333 html = self.convert(text) 

334 

335 # Write to file or stdout 

336 if output: 

337 if isinstance(output, str): 

338 output_file = codecs.open(output, "w", 

339 encoding=encoding, 

340 errors="xmlcharrefreplace") 

341 output_file.write(html) 

342 output_file.close() 

343 else: 

344 writer = codecs.getwriter(encoding) 

345 output_file = writer(output, errors="xmlcharrefreplace") 

346 output_file.write(html) 

347 # Don't close here. User may want to write more. 

348 else: 

349 # Encode manually and write bytes to stdout. 

350 html = html.encode(encoding, "xmlcharrefreplace") 

351 try: 

352 # Write bytes directly to buffer (Python 3). 

353 sys.stdout.buffer.write(html) 

354 except AttributeError: # pragma: no cover 

355 # Probably Python 2, which works with bytes by default. 

356 sys.stdout.write(html) 

357 

358 return self
