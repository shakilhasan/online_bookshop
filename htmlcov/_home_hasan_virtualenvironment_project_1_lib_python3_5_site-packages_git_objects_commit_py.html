282 """Commit the given tree, creating a commit object. 

283 

284 :param repo: Repo object the commit should be part of 

285 :param tree: Tree object or hex or bin sha 

286 the tree of the new commit 

287 :param message: Commit message. It may be an empty string if no message is provided. 

288 It will be converted to a string in any case. 

289 :param parent_commits: 

290 Optional Commit objects to use as parents for the new commit. 

291 If empty list, the commit will have no parents at all and become 

292 a root commit. 

293 If None , the current head commit will be the parent of the 

294 new commit object 

295 :param head: 

296 If True, the HEAD will be advanced to the new commit automatically. 

297 Else the HEAD will remain pointing on the previous commit. This could 

298 lead to undesired results when diffing files. 

299 :param author: The name of the author, optional. If unset, the repository 

300 configuration is used to obtain this value. 

301 :param committer: The name of the committer, optional. If unset, the 

302 repository configuration is used to obtain this value. 

303 :param author_date: The timestamp for the author field 

304 :param commit_date: The timestamp for the committer field 

305

314 parent_commits = [repo.head.commit] 

315 except ValueError: 

316 # empty repositories have no head commit 

317 parent_commits = [] 

318 # END handle parent commits 

319 else: 

320 for p in parent_commits: 

321 if not isinstance(p, cls): 

322 raise ValueError("Parent commit '%r' must be of type %s" % (p, cls))

324 # END if parent commits are unset

347 elif author_date_str:

349 else:

404 def _serialize(self, stream): 

405 write = stream.write 

406 write(("tree %s\n" % self.tree).encode('ascii')) 

407 for p in self.parents: 

408 write(("parent %s\n" % p).encode('ascii')) 

409 

410 a = self.author 

411 aname = a.name 

412 c = self.committer 

413 fmt = "%s %s <%s> %s %s\n" 

414 write((fmt % ("author", aname, a.email, 

415 self.authored_date, 

416 altz_to_utctz_str(self.author_tz_offset))).encode(self.encoding)) 

417 

418 # encode committer 

419 aname = c.name 

420 write((fmt % ("committer", aname, c.email, 

421 self.committed_date, 

422 altz_to_utctz_str(self.committer_tz_offset))).encode(self.encoding)) 

423 

424 if self.encoding != self.default_encoding: 

425 write(("encoding %s\n" % self.encoding).encode('ascii')) 

426 

427 try: 

428 if self.__getattribute__('gpgsig') is not None: 

429 write(b"gpgsig") 

430 for sigline in self.gpgsig.rstrip("\n").split("\n"): 

431 write((" " + sigline + "\n").encode('ascii')) 

432 except AttributeError: 

433 pass 

434 

435 write(b"\n") 

436 

437 # write plain bytes, be sure its encoded according to our encoding 

438 if isinstance(self.message, str): 

439 write(self.message.encode(self.encoding)) 

440 else: 

441 write(self.message) 

442 # END handle encoding 

443 return self

445 def _deserialize(self, stream): 

446 """:param from_rev_list: if true, the stream format is coming from the rev-list command 

447 Otherwise it is assumed to be a plain data stream from our object""" 

448 readline = stream.readline 

449 self.tree = Tree(self.repo, hex_to_bin(readline().split()[1]), Tree.tree_id << 12, '') 

450 

451 self.parents = [] 

452 next_line = None 

453 while True: 

454 parent_line = readline() 

455 if not parent_line.startswith(b'parent'): 

456 next_line = parent_line 

457 break 

458 # END abort reading parents 

459 self.parents.append(type(self)(self.repo, hex_to_bin(parent_line.split()[-1].decode('ascii')))) 

460 # END for each parent line 

461 self.parents = tuple(self.parents) 

462 

463 # we don't know actual author encoding before we have parsed it, so keep the lines around 

464 author_line = next_line 

465 committer_line = readline() 

466 

467 # we might run into one or more mergetag blocks, skip those for now 

468 next_line = readline() 

469 while next_line.startswith(b'mergetag '): 

470 next_line = readline() 

471 while next_line.startswith(b' '): 

472 next_line = readline() 

473 # end skip mergetags 

474 

475 # now we can have the encoding line, or an empty line followed by the optional 

476 # message. 

477 self.encoding = self.default_encoding 

478 self.gpgsig = None 

479 

480 # read headers 

481 enc = next_line 

482 buf = enc.strip() 

483 while buf: 

484 if buf[0:10] == b"encoding ": 

485 self.encoding = buf[buf.find(' ') + 1:].decode( 

486 self.encoding, 'ignore') 

487 elif buf[0:7] == b"gpgsig ": 

488 sig = buf[buf.find(b' ') + 1:] + b"\n" 

489 is_next_header = False 

490 while True: 

491 sigbuf = readline() 

492 if not sigbuf: 

493 break 

494 if sigbuf[0:1] != b" ": 

495 buf = sigbuf.strip() 

496 is_next_header = True 

497 break 

498 sig += sigbuf[1:] 

499 # end read all signature 

500 self.gpgsig = sig.rstrip(b"\n").decode(self.encoding, 'ignore') 

501 if is_next_header: 

502 continue 

503 buf = readline().strip() 

504 # decode the authors name 

505 

506 try: 

507 self.author, self.authored_date, self.author_tz_offset = \ 

508 parse_actor_and_date(author_line.decode(self.encoding, 'replace')) 

509 except UnicodeDecodeError: 

510 log.error("Failed to decode author line '%s' using encoding %s", author_line, self.encoding, 

511 exc_info=True) 

512 

513 try: 

514 self.committer, self.committed_date, self.committer_tz_offset = \ 

515 parse_actor_and_date(committer_line.decode(self.encoding, 'replace')) 

516 except UnicodeDecodeError: 

517 log.error("Failed to decode committer line '%s' using encoding %s", committer_line, self.encoding, 

518 exc_info=True) 

519 # END handle author's encoding 

520 

521 # a stream from our data simply gives us the plain message 

522 # The end of our message stream is marked with a newline that we strip 

523 self.message = stream.read() 

524 try: 

525 self.message = self.message.decode(self.encoding, 'replace') 

526 except UnicodeDecodeError: 

527 log.error("Failed to decode message '%s' using encoding %s", self.message, self.encoding, exc_info=True) 

528 # END exception handling 

529 

530 return self
