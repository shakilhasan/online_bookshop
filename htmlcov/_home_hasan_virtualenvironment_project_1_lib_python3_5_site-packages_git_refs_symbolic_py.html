85 @classmethod 

86 def _iter_packed_refs(cls, repo): 

87 """Returns an iterator yielding pairs of sha1/path pairs (as bytes) for the corresponding refs. 

88 :note: The packed refs file will be kept open as long as we iterate""" 

89 try: 

90 with open(cls._get_packed_refs_path(repo), 'rt') as fp: 

91 for line in fp: 

92 line = line.strip() 

93 if not line: 

94 continue 

95 if line.startswith('#'): 

96 # "# pack-refs with: peeled fully-peeled sorted" 

97 # the git source code shows "peeled", 

98 # "fully-peeled" and "sorted" as the keywords 

99 # that can go on this line, as per comments in git file 

100 # refs/packed-backend.c 

101 # I looked at master on 2017-10-11, 

102 # commit 111ef79afe, after tag v2.15.0-rc1 

103 # from repo https://github.com/git/git.git 

104 if line.startswith('# pack-refs with:') and 'peeled' not in line: 

105 raise TypeError("PackingType of packed-Refs not understood: %r" % line) 

106 # END abort if we do not understand the packing scheme 

107 continue 

108 # END parse comment 

109 

110 # skip dereferenced tag object entries - previous line was actual 

111 # tag reference for it 

112 if line[0] == '^': 

113 continue 

114 

115 yield tuple(line.split(' ', 1)) 

116 # END for each line 

117 except (OSError, IOError): 

118 return 

119 # END no packed-refs file handling 

120 # NOTE: Had try-finally block around here to close the fp, 

121 # but some python version wouldn't allow yields within that. 

122 # I believe files are closing themselves on destruction, so it is 

123 # alright.

137 @classmethod 

138 def _get_ref_info_helper(cls, repo, ref_path): 

139 """Return: (str(sha), str(target_ref_path)) if available, the sha the file at 

140 rela_path points to, or None. target_ref_path is the reference we 

141 point to, or None""" 

142 tokens = None 

143 repodir = _git_dir(repo, ref_path) 

144 try: 

145 with open(osp.join(repodir, ref_path), 'rt', encoding='UTF-8') as fp: 

146 value = fp.read().rstrip() 

147 # Don't only split on spaces, but on whitespace, which allows to parse lines like 

148 # 60b64ef992065e2600bfef6187a97f92398a9144 branch 'master' of git-server:/path/to/repo 

149 tokens = value.split() 

150 assert(len(tokens) != 0) 

151 except (OSError, IOError): 

152 # Probably we are just packed, find our entry in the packed refs file 

153 # NOTE: We are not a symbolic ref if we are in a packed file, as these 

154 # are excluded explicitly 

155 for sha, path in cls._iter_packed_refs(repo): 

156 if path != ref_path: 

157 continue 

158 # sha will be used 

159 tokens = sha, path 

160 break 

161 # END for each packed ref

163 if tokens is None: 

164 raise ValueError("Reference at %r does not exist" % ref_path)

168 return (None, tokens[1]) 

169

172 return (tokens[0], None)

275 def set_reference(self, ref, logmsg=None): 

276 """Set ourselves to the given ref. It will stay a symbol if the ref is a Reference. 

277 Otherwise an Object, given as Object instance or refspec, is assumed and if valid, 

278 will be set which effectively detaches the refererence if it was a purely 

279 symbolic one. 

280 

281 :param ref: SymbolicReference instance, Object instance or refspec string 

282 Only if the ref is a SymbolicRef instance, we will point to it. Everything 

283 else is dereferenced to obtain the actual object. 

284 :param logmsg: If set to a string, the message will be used in the reflog. 

285 Otherwise, a reflog entry is not written for the changed reference. 

286 The previous commit of the entry will be the commit we point to now. 

287 

288 See also: log_append() 

289 

290 :return: self 

291 :note: This symbolic reference will not be dereferenced. For that, see 

292 ``set_object(...)``""" 

293 write_value = None 

294 obj = None 

295 if isinstance(ref, SymbolicReference): 

296 write_value = "ref: %s" % ref.path 

297 elif isinstance(ref, Object): 

298 obj = ref 

299 write_value = ref.hexsha 

300 elif isinstance(ref, str): 

301 try: 

302 obj = self.repo.rev_parse(ref + "^{}") # optionally deref tags 

303 write_value = obj.hexsha 

304 except (BadObject, BadName) as e: 

305 raise ValueError("Could not extract object from %s" % ref) from e 

306 # END end try string 

307 else: 

308 raise ValueError("Unrecognized Value: %r" % ref) 

309 # END try commit attribute 

310 

311 # typecheck 

312 if obj is not None and self._points_to_commits_only and obj.type != Commit.type: 

313 raise TypeError("Require commit, got %r" % obj) 

314 # END verify type 

315 

316 oldbinsha = None 

317 if logmsg is not None: 

318 try: 

319 oldbinsha = self.commit.binsha 

320 except ValueError: 

321 oldbinsha = Commit.NULL_BIN_SHA 

322 # END handle non-existing 

323 # END retrieve old hexsha 

324 

325 fpath = self.abspath 

326 assure_directory_exists(fpath, is_file=True) 

327 

328 lfd = LockedFD(fpath) 

329 fd = lfd.open(write=True, stream=True) 

330 ok = True 

331 try: 

332 fd.write(write_value.encode('ascii') + b'\n') 

333 lfd.commit() 

334 ok = True 

335 finally: 

336 if not ok: 

337 lfd.rollback() 

338 # Adjust the reflog 

339 if logmsg is not None: 

340 self.log_append(oldbinsha, logmsg) 

341 

342 return self

424 @classmethod

426 """Delete the reference at the given path 

427 

428 :param repo: 

429 Repository to delete the reference from 

430 

431 :param path: 

432 Short or full path pointing to the reference, i.e. refs/myreference 

433 or just "myreference", hence 'refs/' is implied. 

434 Alternatively the symbolic reference to be deleted""" 

435 full_ref_path = cls.to_full_path(path) 

436 abs_path = osp.join(repo.common_dir, full_ref_path) 

437 if osp.exists(abs_path): 

438 os.remove(abs_path) 

439 else: 

440 # check packed refs 

441 pack_file_path = cls._get_packed_refs_path(repo) 

442 try: 

443 with open(pack_file_path, 'rb') as reader: 

444 new_lines = [] 

445 made_change = False 

446 dropped_last_line = False 

447 for line in reader: 

448 line = line.decode(defenc) 

449 _, _, line_ref = line.partition(' ') 

450 line_ref = line_ref.strip() 

451 # keep line if it is a comment or if the ref to delete is not 

452 # in the line 

453 # If we deleted the last line and this one is a tag-reference object, 

454 # we drop it as well 

455 if (line.startswith('#') or full_ref_path != line_ref) and \ 

456 (not dropped_last_line or dropped_last_line and not line.startswith('^')): 

457 new_lines.append(line) 

458 dropped_last_line = False 

459 continue 

460 # END skip comments and lines without our path 

461 

462 # drop this line 

463 made_change = True 

464 dropped_last_line = True 

465 

466 # write the new lines 

467 if made_change: 

468 # write-binary is required, otherwise windows will 

469 # open the file in text mode and change LF to CRLF ! 

470 with open(pack_file_path, 'wb') as fd: 

471 fd.writelines(line.encode(defenc) for line in new_lines) 

472 

473 except (OSError, IOError): 

474 pass # it didn't exist at all 

475 

476 # delete the reflog 

477 reflog_path = RefLog.path(cls(repo, full_ref_path)) 

478 if osp.isfile(reflog_path): 

479 os.remove(reflog_path) 

480 # END remove reflog

547 def rename(self, new_path, force=False): 

548 """Rename self to a new path 

549 

550 :param new_path: 

551 Either a simple name or a full path, i.e. new_name or features/new_name. 

552 The prefix refs/ is implied for references and will be set as needed. 

553 In case this is a symbolic ref, there is no implied prefix

561 new_path = self.to_full_path(new_path) 

562 if self.path == new_path: 

563 return self

567 if osp.isfile(new_abs_path): 

568 if not force: 

569 # if they point to the same file, its not an error 

570 with open(new_abs_path, 'rb') as fd1: 

571 f1 = fd1.read().strip() 

572 with open(cur_abs_path, 'rb') as fd2: 

573 f2 = fd2.read().strip() 

574 if f1 != f2:

577 # but clarity we just continue as usual 

578 # END not force handling 

579 os.remove(new_abs_path)
