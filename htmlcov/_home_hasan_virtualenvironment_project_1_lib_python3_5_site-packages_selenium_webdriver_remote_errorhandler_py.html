103 def check_response(self, response): 

104 """ 

105 Checks that a JSON response from the WebDriver does not have an error. 

106 

107 :Args: 

108 - response - The JSON response from the WebDriver server as a dictionary 

109 object. 

110 

111 :Raises: If the response contains an error message. 

112 """ 

113 status = response.get('status', None) 

114 if status is None or status == ErrorCode.SUCCESS: 

115 return 

116 value = None 

117 message = response.get("message", "") 

118 screen = response.get("screen", "") 

119 stacktrace = None 

120 if isinstance(status, int): 

121 value_json = response.get('value', None) 

122 if value_json and isinstance(value_json, basestring): 

123 import json 

124 try: 

125 value = json.loads(value_json) 

126 if len(value.keys()) == 1: 

127 value = value['value'] 

128 status = value.get('error', None) 

129 if status is None: 

130 status = value["status"] 

131 message = value["value"] 

132 if not isinstance(message, basestring): 

133 value = message 

134 message = message.get('message') 

135 else: 

136 message = value.get('message', None) 

137 except ValueError: 

138 pass
