41def _clean_credentials(credentials): 

42 """ 

43 Clean a dictionary of credentials of potentially sensitive info before 

44 sending to less secure functions. 

45 

46 Not comprehensive - intended for user_login_failed signal 

47 """ 

48 SENSITIVE_CREDENTIALS = re.compile('api|token|key|secret|password|signature', re.I) 

49 CLEANSED_SUBSTITUTE = '********************' 

50 for key in credentials: 

51 if SENSITIVE_CREDENTIALS.search(key): 

52 credentials[key] = CLEANSED_SUBSTITUTE 

53 return credentials

62def authenticate(request=None, **credentials): 

63 """ 

64 If the given credentials are valid, return a User object. 

65 """ 

66 for backend, backend_path in _get_backends(return_tuples=True): 

67 try: 

68 inspect.getcallargs(backend.authenticate, request, **credentials) 

69 except TypeError: 

70 # This backend doesn't accept these credentials as arguments. Try the next one. 

71 continue 

72 try: 

73 user = backend.authenticate(request, **credentials) 

74 except PermissionDenied: 

75 # This backend says to stop in our tracks - this user should not be allowed in at all. 

76 break 

77 if user is None: 

78 continue 

79 # Annotate the user object with the path of the backend. 

80 user.backend = backend_path 

81 return user 

82 

83 # The credentials supplied are invalid to all backends, fire signal 

84 user_login_failed.send(sender=__name__, credentials=_clean_credentials(credentials), request=request)

87def login(request, user, backend=None): 

88 """ 

89 Persist a user id and a backend in the request. This way a user doesn't 

90 have to reauthenticate on every request. Note that data set during 

91 the anonymous session is retained when the user logs in. 

92 """ 

93 session_auth_hash = '' 

94 if user is None: 

95 user = request.user 

96 if hasattr(user, 'get_session_auth_hash'): 

97 session_auth_hash = user.get_session_auth_hash() 

98 

99 if SESSION_KEY in request.session: 

100 if _get_user_session_key(request) != user.pk or ( 

101 session_auth_hash and 

102 not constant_time_compare(request.session.get(HASH_SESSION_KEY, ''), session_auth_hash)): 

103 # To avoid reusing another user's session, create a new, empty 

104 # session if the existing session corresponds to a different 

105 # authenticated user. 

106 request.session.flush() 

107 else: 

108 request.session.cycle_key() 

109 

110 try: 

111 backend = backend or user.backend 

112 except AttributeError: 

113 backends = _get_backends(return_tuples=True) 

114 if len(backends) == 1: 

115 _, backend = backends[0] 

116 else: 

117 raise ValueError( 

118 'You have multiple authentication backends configured and ' 

119 'therefore must provide the `backend` argument or set the ' 

120 '`backend` attribute on the user.' 

121 ) 

122 else: 

123 if not isinstance(backend, str): 

124 raise TypeError('backend must be a dotted import path string (got %r).' % backend) 

125 

126 request.session[SESSION_KEY] = user._meta.pk.value_to_string(user) 

127 request.session[BACKEND_SESSION_KEY] = backend 

128 request.session[HASH_SESSION_KEY] = session_auth_hash 

129 if hasattr(request, 'user'): 

130 request.user = user 

131 rotate_token(request) 

132 user_logged_in.send(sender=user.__class__, request=request, user=user)

136 """

138 data. 

139 """ 

140 # Dispatch the signal before the user is logged out so the receivers have a 

141 # chance to find out *who* logged out. 

142 user = getattr(request, 'user', None)

144 user = None 

145 user_logged_out.send(sender=user.__class__, request=request, user=user)

174def get_user(request): 

175 """ 

176 Return the user model instance associated with the given request session. 

177 If no user is retrieved, return an instance of `AnonymousUser`. 

178 """ 

179 from .models import AnonymousUser 

180 user = None 

181 try: 

182 user_id = _get_user_session_key(request) 

183 backend_path = request.session[BACKEND_SESSION_KEY] 

184 except KeyError: 

185 pass 

186 else: 

187 if backend_path in settings.AUTHENTICATION_BACKENDS: 

188 backend = load_backend(backend_path) 

189 user = backend.get_user(user_id) 

190 # Verify the session 

191 if hasattr(user, 'get_session_auth_hash'): 

192 session_hash = request.session.get(HASH_SESSION_KEY) 

193 session_hash_verified = session_hash and constant_time_compare( 

194 session_hash, 

195 user.get_session_auth_hash() 

196 ) 

197 if not session_hash_verified: 

198 request.session.flush() 

199 user = None 

200 

201 return user or AnonymousUser()
