47class DecompressMemMapReader(LazyMixin): 

48 

49 """Reads data in chunks from a memory map and decompresses it. The client sees 

50 only the uncompressed data, respective file-like read calls are handling on-demand 

51 buffered decompression accordingly 

52 

53 A constraint on the total size of bytes is activated, simulating 

54 a logical file within a possibly larger physical memory area 

55 

56 To read efficiently, you clearly don't want to read individual bytes, instead, 

57 read a few kilobytes at least. 

58 

59 **Note:** The chunk-size should be carefully selected as it will involve quite a bit 

60 of string copying due to the way the zlib is implemented. Its very wasteful, 

61 hence we try to find a good tradeoff between allocation time and number of 

62 times we actually allocate. An own zlib implementation would be good here 

63 to better support streamed reading - it would only need to keep the mmap 

64 and decompress it into chunks, that's all ... """ 

65 __slots__ = ('_m', '_zip', '_buf', '_buflen', '_br', '_cws', '_cwe', '_s', '_close', 

66 '_cbr', '_phi')

95 def _parse_header_info(self): 

96 """If this stream contains object data, parse the header info and skip the 

97 stream to a point where each read will yield object content 

98 

99 :return: parsed type_string, size""" 

100 # read header 

101 # should really be enough, cgit uses 8192 I believe 

102 # And for good reason !! This needs to be that high for the header to be read correctly in all cases 

103 maxb = 8192 

104 self._s = maxb 

105 hdr = self.read(maxb) 

106 hdrend = hdr.find(NULL_BYTE) 

107 typ, size = hdr[:hdrend].split(BYTE_SPACE) 

108 size = int(size) 

109 self._s = size 

110 

111 # adjust internal state to match actual header length that we ignore 

112 # The buffer will be depleted first on future reads 

113 self._br = 0 

114 hdrend += 1 

115 self._buf = BytesIO(hdr[hdrend:]) 

116 self._buflen = len(hdr) - hdrend 

117 

118 self._phi = True 

119 

120 return typ, size

152 def compressed_bytes_read(self): 

153 """ 

154 :return: number of compressed bytes read. This includes the bytes it 

155 took to decompress the header ( if there was one )""" 

156 # ABSTRACT: When decompressing a byte stream, it can be that the first 

157 # x bytes which were requested match the first x bytes in the loosely 

158 # compressed datastream. This is the worst-case assumption that the reader 

159 # does, it assumes that it will get at least X bytes from X compressed bytes 

160 # in call cases. 

161 # The caveat is that the object, according to our known uncompressed size, 

162 # is already complete, but there are still some bytes left in the compressed 

163 # stream that contribute to the amount of compressed bytes. 

164 # How can we know that we are truly done, and have read all bytes we need 

165 # to read ? 

166 # Without help, we cannot know, as we need to obtain the status of the 

167 # decompression. If it is not finished, we need to decompress more data 

168 # until it is finished, to yield the actual number of compressed bytes 

169 # belonging to the decompressed object 

170 # We are using a custom zlib module for this, if its not present, 

171 # we try to put in additional bytes up for decompression if feasible 

172 # and check for the unused_data. 

173 

174 # Only scrub the stream forward if we are officially done with the 

175 # bytes we were to have. 

176 if self._br == self._s and not self._zip.unused_data: 

177 # manipulate the bytes-read to allow our own read method to continue 

178 # but keep the window at its current position 

179 self._br = 0 

180 if hasattr(self._zip, 'status'): 

181 while self._zip.status == zlib.Z_OK: 

182 self.read(mmap.PAGESIZE) 

183 # END scrub-loop custom zlib 

184 else: 

185 # pass in additional pages, until we have unused data 

186 while not self._zip.unused_data and self._cbr != len(self._m): 

187 self.read(mmap.PAGESIZE) 

188 # END scrub-loop default zlib 

189 # END handle stream scrubbing 

190 

191 # reset bytes read, just to be sure 

192 self._br = self._s 

193 # END handle stream scrubbing 

194 

195 # unused data ends up in the unconsumed tail, which was removed 

196 # from the count already 

197 return self._cbr

201 def seek(self, offset, whence=getattr(os, 'SEEK_SET', 0)): 

202 """Allows to reset the stream to restart reading 

203 :raise ValueError: If offset and whence are not 0""" 

204 if offset != 0 or whence != getattr(os, 'SEEK_SET', 0): 

205 raise ValueError("Can only seek to position 0") 

206 # END handle offset

208 self._zip = zlib.decompressobj() 

209 self._br = self._cws = self._cwe = self._cbr = 0 

210 if self._phi: 

211 self._phi = False 

212 del(self._s) # trigger header parsing on first access

215 def read(self, size=-1): 

216 if size < 1: 

217 size = self._s - self._br 

218 else: 

219 size = min(size, self._s - self._br) 

220 # END clamp size 

221 

222 if size == 0: 

223 return bytes() 

224 # END handle depletion 

225 

226 # deplete the buffer, then just continue using the decompress object 

227 # which has an own buffer. We just need this to transparently parse the 

228 # header from the zlib stream 

229 dat = bytes() 

230 if self._buf: 

231 if self._buflen >= size: 

232 # have enough data 

233 dat = self._buf.read(size) 

234 self._buflen -= size 

235 self._br += size 

236 return dat 

237 else: 

238 dat = self._buf.read() # ouch, duplicates data 

239 size -= self._buflen 

240 self._br += self._buflen 

241 

242 self._buflen = 0 

243 self._buf = None 

244 # END handle buffer len 

245 # END handle buffer 

246 

247 # decompress some data 

248 # Abstract: zlib needs to operate on chunks of our memory map ( which may 

249 # be large ), as it will otherwise and always fill in the 'unconsumed_tail' 

250 # attribute which possible reads our whole map to the end, forcing 

251 # everything to be read from disk even though just a portion was requested. 

252 # As this would be a nogo, we workaround it by passing only chunks of data, 

253 # moving the window into the memory map along as we decompress, which keeps 

254 # the tail smaller than our chunk-size. This causes 'only' the chunk to be 

255 # copied once, and another copy of a part of it when it creates the unconsumed 

256 # tail. We have to use it to hand in the appropriate amount of bytes during 

257 # the next read. 

258 tail = self._zip.unconsumed_tail 

259 if tail: 

260 # move the window, make it as large as size demands. For code-clarity, 

261 # we just take the chunk from our map again instead of reusing the unconsumed 

262 # tail. The latter one would safe some memory copying, but we could end up 

263 # with not getting enough data uncompressed, so we had to sort that out as well. 

264 # Now we just assume the worst case, hence the data is uncompressed and the window 

265 # needs to be as large as the uncompressed bytes we want to read. 

266 self._cws = self._cwe - len(tail) 

267 self._cwe = self._cws + size 

268 else: 

269 cws = self._cws 

270 self._cws = self._cwe 

271 self._cwe = cws + size 

272 # END handle tail 

273 

274 # if window is too small, make it larger so zip can decompress something 

275 if self._cwe - self._cws < 8: 

276 self._cwe = self._cws + 8 

277 # END adjust winsize 

278 

279 # takes a slice, but doesn't copy the data, it says ... 

280 indata = self._m[self._cws:self._cwe] 

281 

282 # get the actual window end to be sure we don't use it for computations 

283 self._cwe = self._cws + len(indata) 

284 dcompdat = self._zip.decompress(indata, size) 

285 # update the amount of compressed bytes read 

286 # We feed possibly overlapping chunks, which is why the unconsumed tail 

287 # has to be taken into consideration, as well as the unused data 

288 # if we hit the end of the stream 

289 # NOTE: Behavior changed in PY2.7 onward, which requires special handling to make the tests work properly. 

290 # They are thorough, and I assume it is truly working. 

291 # Why is this logic as convoluted as it is ? Please look at the table in  

292 # https://github.com/gitpython-developers/gitdb/issues/19 to learn about the test-results. 

293 # Bascially, on py2.6, you want to use branch 1, whereas on all other python version, the second branch 

294 # will be the one that works.  

295 # However, the zlib VERSIONs as well as the platform check is used to further match the entries in the  

296 # table in the github issue. This is it ... it was the only way I could make this work everywhere. 

297 # IT's CERTAINLY GOING TO BITE US IN THE FUTURE ... . 

298 if PY26 or ((zlib.ZLIB_VERSION == '1.2.7' or zlib.ZLIB_VERSION == '1.2.5') and not sys.platform == 'darwin'): 

299 unused_datalen = len(self._zip.unconsumed_tail) 

300 else: 

301 unused_datalen = len(self._zip.unconsumed_tail) + len(self._zip.unused_data) 

302 # # end handle very special case ... 

303 

304 self._cbr += len(indata) - unused_datalen 

305 self._br += len(dcompdat) 

306 

307 if dat: 

308 dcompdat = dat + dcompdat 

309 # END prepend our cached data 

310 

311 # it can happen, depending on the compression, that we get less bytes 

312 # than ordered as it needs the final portion of the data as well. 

313 # Recursively resolve that. 

314 # Note: dcompdat can be empty even though we still appear to have bytes 

315 # to read, if we are called by compressed_bytes_read - it manipulates 

316 # us to empty the stream 

317 if dcompdat and (len(dcompdat) - len(dat)) < size and self._br < self._s: 

318 dcompdat += self.read(size - len(dcompdat)) 

319 # END handle special case 

320 return dcompdat

323class DeltaApplyReader(LazyMixin): 

324 

325 """A reader which dynamically applies pack deltas to a base object, keeping the 

326 memory demands to a minimum. 

327 

328 The size of the final object is only obtainable once all deltas have been 

329 applied, unless it is retrieved from a pack index. 

330 

331 The uncompressed Delta has the following layout (MSB being a most significant 

332 bit encoded dynamic size): 

333 

334 * MSB Source Size - the size of the base against which the delta was created 

335 * MSB Target Size - the size of the resulting data after the delta was applied 

336 * A list of one byte commands (cmd) which are followed by a specific protocol: 

337 

338 * cmd & 0x80 - copy delta_data[offset:offset+size] 

339 

340 * Followed by an encoded offset into the delta data 

341 * Followed by an encoded size of the chunk to copy 

342 

343 * cmd & 0x7f - insert 

344 

345 * insert cmd bytes from the delta buffer into the output stream 

346 

347 * cmd == 0 - invalid operation ( or error in delta stream ) 

348 """ 

349 __slots__ = ( 

350 "_bstream", # base stream to which to apply the deltas 

351 "_dstreams", # tuple of delta stream readers 

352 "_mm_target", # memory map of the delta-applied data 

353 "_size", # actual number of bytes in _mm_target 

354 "_br" # number of bytes read 

355 ) 

356 

357 #{ Configuration 

358 k_max_memory_move = 250 * 1000 * 1000 

359 #} END configuration 

360 

361 def __init__(self, stream_list): 

362 """Initialize this instance with a list of streams, the first stream being 

363 the delta to apply on top of all following deltas, the last stream being the 

364 base object onto which to apply the deltas""" 

365 assert len(stream_list) > 1, "Need at least one delta and one base stream" 

366 

367 self._bstream = stream_list[-1] 

368 self._dstreams = tuple(stream_list[:-1]) 

369 self._br = 0 

370 

371 def _set_cache_too_slow_without_c(self, attr): 

372 # the direct algorithm is fastest and most direct if there is only one 

373 # delta. Also, the extra overhead might not be worth it for items smaller 

374 # than X - definitely the case in python, every function call costs 

375 # huge amounts of time 

376 # if len(self._dstreams) * self._bstream.size < self.k_max_memory_move: 

377 if len(self._dstreams) == 1: 

378 return self._set_cache_brute_(attr) 

379 

380 # Aggregate all deltas into one delta in reverse order. Hence we take 

381 # the last delta, and reverse-merge its ancestor delta, until we receive 

382 # the final delta data stream. 

383 dcl = connect_deltas(self._dstreams) 

384 

385 # call len directly, as the (optional) c version doesn't implement the sequence 

386 # protocol 

387 if dcl.rbound() == 0: 

388 self._size = 0 

389 self._mm_target = allocate_memory(0) 

390 return 

391 # END handle empty list 

392 

393 self._size = dcl.rbound() 

394 self._mm_target = allocate_memory(self._size) 

395 

396 bbuf = allocate_memory(self._bstream.size) 

397 stream_copy(self._bstream.read, bbuf.write, self._bstream.size, 256 * mmap.PAGESIZE) 

398 

399 # APPLY CHUNKS 

400 write = self._mm_target.write 

401 dcl.apply(bbuf, write) 

402 

403 self._mm_target.seek(0) 

404 

405 def _set_cache_brute_(self, attr): 

406 """If we are here, we apply the actual deltas""" 

407 # TODO: There should be a special case if there is only one stream 

408 # Then the default-git algorithm should perform a tad faster, as the 

409 # delta is not peaked into, causing less overhead. 

410 buffer_info_list = list() 

411 max_target_size = 0 

412 for dstream in self._dstreams: 

413 buf = dstream.read(512) # read the header information + X 

414 offset, src_size = msb_size(buf) 

415 offset, target_size = msb_size(buf, offset) 

416 buffer_info_list.append((buf[offset:], offset, src_size, target_size)) 

417 max_target_size = max(max_target_size, target_size) 

418 # END for each delta stream 

419 

420 # sanity check - the first delta to apply should have the same source 

421 # size as our actual base stream 

422 base_size = self._bstream.size 

423 target_size = max_target_size 

424 

425 # if we have more than 1 delta to apply, we will swap buffers, hence we must 

426 # assure that all buffers we use are large enough to hold all the results 

427 if len(self._dstreams) > 1: 

428 base_size = target_size = max(base_size, max_target_size) 

429 # END adjust buffer sizes 

430 

431 # Allocate private memory map big enough to hold the first base buffer 

432 # We need random access to it 

433 bbuf = allocate_memory(base_size) 

434 stream_copy(self._bstream.read, bbuf.write, base_size, 256 * mmap.PAGESIZE) 

435 

436 # allocate memory map large enough for the largest (intermediate) target 

437 # We will use it as scratch space for all delta ops. If the final 

438 # target buffer is smaller than our allocated space, we just use parts 

439 # of it upon return. 

440 tbuf = allocate_memory(target_size) 

441 

442 # for each delta to apply, memory map the decompressed delta and 

443 # work on the op-codes to reconstruct everything. 

444 # For the actual copying, we use a seek and write pattern of buffer 

445 # slices. 

446 final_target_size = None 

447 for (dbuf, offset, src_size, target_size), dstream in zip(reversed(buffer_info_list), reversed(self._dstreams)): 

448 # allocate a buffer to hold all delta data - fill in the data for 

449 # fast access. We do this as we know that reading individual bytes 

450 # from our stream would be slower than necessary ( although possible ) 

451 # The dbuf buffer contains commands after the first two MSB sizes, the 

452 # offset specifies the amount of bytes read to get the sizes. 

453 ddata = allocate_memory(dstream.size - offset) 

454 ddata.write(dbuf) 

455 # read the rest from the stream. The size we give is larger than necessary 

456 stream_copy(dstream.read, ddata.write, dstream.size, 256 * mmap.PAGESIZE) 

457 

458 ####################################################################### 

459 if 'c_apply_delta' in globals(): 

460 c_apply_delta(bbuf, ddata, tbuf) 

461 else: 

462 apply_delta_data(bbuf, src_size, ddata, len(ddata), tbuf.write) 

463 ####################################################################### 

464 

465 # finally, swap out source and target buffers. The target is now the 

466 # base for the next delta to apply 

467 bbuf, tbuf = tbuf, bbuf 

468 bbuf.seek(0) 

469 tbuf.seek(0) 

470 final_target_size = target_size 

471 # END for each delta to apply 

472 

473 # its already seeked to 0, constrain it to the actual size 

474 # NOTE: in the end of the loop, it swaps buffers, hence our target buffer 

475 # is not tbuf, but bbuf ! 

476 self._mm_target = bbuf 

477 self._size = final_target_size 

478 

479 #{ Configuration 

480 if not has_perf_mod: 

481 _set_cache_ = _set_cache_brute_ 

482 else: 

483 _set_cache_ = _set_cache_too_slow_without_c 

484 

485 #} END configuration 

486 

487 def read(self, count=0): 

488 bl = self._size - self._br # bytes left 

489 if count < 1 or count > bl: 

490 count = bl 

491 # NOTE: we could check for certain size limits, and possibly 

492 # return buffers instead of strings to prevent byte copying 

493 data = self._mm_target.read(count) 

494 self._br += len(data) 

495 return data

501 if offset != 0 or whence != getattr(os, 'SEEK_SET', 0): 

502 raise ValueError("Can only seek to position 0") 

503 # END handle offset 

504 self._br = 0 

505 self._mm_target.seek(0)
