29class Selector(object): 

30 """Core test selector. Examines test candidates and determines whether, 

31 given the specified configuration, the test candidate should be selected 

32 as a test. 

33 """ 

34 def __init__(self, config): 

35 if config is None: 

36 config = Config() 

37 self.configure(config) 

38 

39 def configure(self, config): 

40 self.config = config 

41 self.exclude = config.exclude 

42 self.ignoreFiles = config.ignoreFiles 

43 self.include = config.include 

44 self.plugins = config.plugins 

45 self.match = config.testMatch 

46 

47 def matches(self, name): 

48 """Does the name match my requirements? 

49 

50 To match, a name must match config.testMatch OR config.include 

51 and it must not match config.exclude 

52 """ 

53 return ((self.match.search(name) 

54 or (self.include and 

55 [_f for _f in [inc.search(name) for inc in self.include] if _f])) 

56 and ((not self.exclude) 

57 or not [_f for _f in [exc.search(name) for exc in self.exclude] if _f] 

58 ))

60 def wantClass(self, cls): 

61 """Is the class a wanted test class? 

62 

63 A class must be a unittest.TestCase subclass, or match test name 

64 requirements. Classes that start with _ are always excluded. 

65 """ 

66 declared = getattr(cls, '__test__', None) 

67 if declared is not None: 

68 wanted = declared 

69 else: 

70 wanted = (not cls.__name__.startswith('_') 

71 and (issubclass(cls, unittest.TestCase) 

72 or self.matches(cls.__name__))) 

73 

74 plug_wants = self.plugins.wantClass(cls) 

75 if plug_wants is not None: 

76 log.debug("Plugin setting selection of %s to %s", cls, plug_wants) 

77 wanted = plug_wants 

78 log.debug("wantClass %s? %s", cls, wanted) 

79 return wanted

81 def wantDirectory(self, dirname): 

82 """Is the directory a wanted test directory? 

83 

84 All package directories match, so long as they do not match exclude.  

85 All other directories must match test requirements. 

86 """ 

87 tail = op_basename(dirname) 

88 if ispackage(dirname): 

89 wanted = (not self.exclude 

90 or not [_f for _f in [exc.search(tail) for exc in self.exclude] if _f]) 

91 else: 

92 wanted = (self.matches(tail) 

93 or (self.config.srcDirs 

94 and tail in self.config.srcDirs)) 

95 plug_wants = self.plugins.wantDirectory(dirname) 

96 if plug_wants is not None: 

97 log.debug("Plugin setting selection of %s to %s", 

98 dirname, plug_wants) 

99 wanted = plug_wants 

100 log.debug("wantDirectory %s? %s", dirname, wanted) 

101 return wanted

103 def wantFile(self, file): 

104 """Is the file a wanted test file? 

105 

106 The file must be a python source file and match testMatch or 

107 include, and not match exclude. Files that match ignore are *never* 

108 wanted, regardless of plugin, testMatch, include or exclude settings. 

109 """ 

110 # never, ever load files that match anything in ignore 

111 # (.* _* and *setup*.py by default) 

112 base = op_basename(file) 

113 ignore_matches = [ ignore_this for ignore_this in self.ignoreFiles 

114 if ignore_this.search(base) ] 

115 if ignore_matches: 

116 log.debug('%s matches ignoreFiles pattern; skipped', 

117 base) 

118 return False 

119 if not self.config.includeExe and is_executable(file): 

120 log.info('%s is executable; skipped', file) 

121 return False 

122 dummy, ext = op_splitext(base) 

123 pysrc = ext == '.py' 

124 

125 wanted = pysrc and self.matches(base) 

126 plug_wants = self.plugins.wantFile(file) 

127 if plug_wants is not None: 

128 log.debug("plugin setting want %s to %s", file, plug_wants) 

129 wanted = plug_wants 

130 log.debug("wantFile %s? %s", file, wanted) 

131 return wanted

133 def wantFunction(self, function): 

134 """Is the function a test function? 

135 """ 

136 try: 

137 if hasattr(function, 'compat_func_name'): 

138 funcname = function.compat_func_name 

139 else: 

140 funcname = function.__name__ 

141 except AttributeError: 

142 # not a function 

143 return False 

144 declared = getattr(function, '__test__', None) 

145 if declared is not None: 

146 wanted = declared 

147 else: 

148 wanted = not funcname.startswith('_') and self.matches(funcname) 

149 plug_wants = self.plugins.wantFunction(function) 

150 if plug_wants is not None: 

151 wanted = plug_wants 

152 log.debug("wantFunction %s? %s", function, wanted) 

153 return wanted

155 def wantMethod(self, method): 

156 """Is the method a test method? 

157 """ 

158 try: 

159 method_name = method.__name__ 

160 except AttributeError: 

161 # not a method 

162 return False 

163 if method_name.startswith('_'): 

164 # never collect 'private' methods 

165 return False 

166 declared = getattr(method, '__test__', None) 

167 if declared is not None: 

168 wanted = declared 

169 else: 

170 wanted = self.matches(method_name) 

171 plug_wants = self.plugins.wantMethod(method) 

172 if plug_wants is not None: 

173 wanted = plug_wants 

174 log.debug("wantMethod %s? %s", method, wanted) 

175 return wanted

177 def wantModule(self, module): 

178 """Is the module a test module? 

179 

180 The tail of the module name must match test requirements. One exception: 

181 we always want __main__. 

182 """ 

183 declared = getattr(module, '__test__', None) 

184 if declared is not None: 

185 wanted = declared 

186 else: 

187 wanted = self.matches(module.__name__.split('.')[-1]) \ 

188 or module.__name__ == '__main__' 

189 plug_wants = self.plugins.wantModule(module) 

190 if plug_wants is not None: 

191 wanted = plug_wants 

192 log.debug("wantModule %s? %s", module, wanted) 

193 return wanted

198class TestAddress(object): 

199 """A test address represents a user's request to run a particular 

200 test. The user may specify a filename or module (or neither), 

201 and/or a callable (a class, function, or method). The naming 

202 format for test addresses is: 

203 

204 filename_or_module:callable 

205 

206 Filenames that are not absolute will be made absolute relative to 

207 the working dir. 

208 

209 The filename or module part will be considered a module name if it 

210 doesn't look like a file, that is, if it doesn't exist on the file 

211 system and it doesn't contain any directory separators and it 

212 doesn't end in .py. 

213 

214 Callables may be a class name, function name, method name, or 

215 class.method specification. 

216 """ 

217 def __init__(self, name, workingDir=None): 

218 if workingDir is None: 

219 workingDir = os.getcwd() 

220 self.name = name 

221 self.workingDir = workingDir 

222 self.filename, self.module, self.call = split_test_name(name) 

223 log.debug('Test name %s resolved to file %s, module %s, call %s', 

224 name, self.filename, self.module, self.call) 

225 if self.filename is None: 

226 if self.module is not None: 

227 self.filename = getfilename(self.module, self.workingDir) 

228 if self.filename: 

229 self.filename = src(self.filename) 

230 if not op_isabs(self.filename): 

231 self.filename = op_abspath(op_join(workingDir, 

232 self.filename)) 

233 if self.module is None: 

234 self.module = getpackage(self.filename) 

235 log.debug( 

236 'Final resolution of test name %s: file %s module %s call %s', 

237 name, self.filename, self.module, self.call) 

238 

239 def totuple(self): 

240 return (self.filename, self.module, self.call) 

241 

242 def __str__(self): 

243 return self.name

246 return "%s: (%s, %s, %s)" % (self.name, self.filename, 

247 self.module, self.call)
