19def inspect_traceback(tb): 

20 """Inspect a traceback and its frame, returning source for the expression 

21 where the exception was raised, with simple variable replacement performed 

22 and the line on which the exception was raised marked with '>>' 

23 """ 

24 log.debug('inspect traceback %s', tb) 

25 

26 # we only want the innermost frame, where the exception was raised 

27 while tb.tb_next: 

28 tb = tb.tb_next 

29 

30 frame = tb.tb_frame 

31 lines, exc_line = tbsource(tb) 

32 

33 # figure out the set of lines to grab. 

34 inspect_lines, mark_line = find_inspectable_lines(lines, exc_line) 

35 src = StringIO(textwrap.dedent(''.join(inspect_lines))) 

36 exp = Expander(frame.f_locals, frame.f_globals) 

37 

38 while inspect_lines: 

39 try: 

40 for tok in tokenize.generate_tokens(src.readline): 

41 exp(*tok) 

42 except tokenize.TokenError as e: 

43 # this can happen if our inspectable region happens to butt up 

44 # against the end of a construct like a docstring with the closing 

45 # """ on separate line 

46 log.debug("Tokenizer error: %s", e) 

47 inspect_lines.pop(0) 

48 mark_line -= 1 

49 src = StringIO(textwrap.dedent(''.join(inspect_lines))) 

50 exp = Expander(frame.f_locals, frame.f_globals) 

51 continue 

52 break 

53 padded = [] 

54 if exp.expanded_source: 

55 exp_lines = exp.expanded_source.split('\n') 

56 ep = 0 

57 for line in exp_lines: 

58 if ep == mark_line: 

59 padded.append('>> ' + line) 

60 else: 

61 padded.append(' ' + line) 

62 ep += 1 

63 return '\n'.join(padded)

66def tbsource(tb, context=6): 

67 """Get source from a traceback object. 

68 

69 A tuple of two things is returned: a list of lines of context from 

70 the source code, and the index of the current line within that list. 

71 The optional second argument specifies the number of lines of context 

72 to return, which are centered around the current line. 

73 

74 .. Note :: 

75 This is adapted from inspect.py in the python 2.4 standard library,  

76 since a bug in the 2.3 version of inspect prevents it from correctly 

77 locating source lines in a traceback frame. 

78 """ 

79 

80 lineno = tb.tb_lineno 

81 frame = tb.tb_frame 

82 

83 if context > 0: 

84 start = lineno - 1 - context//2 

85 log.debug("lineno: %s start: %s", lineno, start) 

86 

87 try: 

88 lines, dummy = inspect.findsource(frame) 

89 except IOError: 

90 lines, index = [''], 0 

91 else: 

92 all_lines = lines 

93 start = max(start, 1) 

94 start = max(0, min(start, len(lines) - context)) 

95 lines = lines[start:start+context] 

96 index = lineno - 1 - start 

97 

98 # python 2.5 compat: if previous line ends in a continuation, 

99 # decrement start by 1 to match 2.4 behavior  

100 if sys.version_info >= (2, 5) and index > 0: 

101 while lines[index-1].strip().endswith('\\'): 

102 start -= 1 

103 lines = all_lines[start:start+context] 

104 else: 

105 lines, index = [''], 0 

106 log.debug("tbsource lines '''%s''' around index %s", lines, index) 

107 return (lines, index)

110def find_inspectable_lines(lines, pos): 

111 """Find lines in home that are inspectable. 

112  

113 Walk back from the err line up to 3 lines, but don't walk back over 

114 changes in indent level. 

115 

116 Walk forward up to 3 lines, counting \ separated lines as 1. Don't walk 

117 over changes in indent level (unless part of an extended line) 

118 """ 

119 cnt = re.compile(r'\\[\s\n]*$') 

120 df = re.compile(r':[\s\n]*$') 

121 ind = re.compile(r'^(\s*)') 

122 toinspect = [] 

123 home = lines[pos] 

124 home_indent = ind.match(home).groups()[0] 

125 

126 before = lines[max(pos-3, 0):pos] 

127 before.reverse() 

128 after = lines[pos+1:min(pos+4, len(lines))] 

129 

130 for line in before: 

131 if ind.match(line).groups()[0] == home_indent: 

132 toinspect.append(line) 

133 else: 

134 break 

135 toinspect.reverse() 

136 toinspect.append(home) 

137 home_pos = len(toinspect)-1 

138 continued = cnt.search(home) 

139 for line in after: 

140 if ((continued or ind.match(line).groups()[0] == home_indent) 

141 and not df.search(line)): 

142 toinspect.append(line) 

143 continued = cnt.search(line) 

144 else: 

145 break 

146 log.debug("Inspecting lines '''%s''' around %s", toinspect, home_pos) 

147 return toinspect, home_pos
