131 """ 

132 Supports 'iso-8601' date format too which is out the scope of 

133 the ``datetime.strptime`` standard library 

134 

135 # ISO 8601: ``http://www.w3.org/TR/NOTE-datetime``

153class BaseCSVField(forms.Field): 

154 """ 

155 Base field for validating CSV types. Value validation is performed by 

156 secondary base classes. 

157 

158 ex:: 

159 class IntegerCSVField(BaseCSVField, filters.IntegerField): 

160 pass

169 super().__init__(*args, **kwargs)

171 def _get_widget_class(self, widget): 

172 # passthrough, allows for override 

173 if isinstance(widget, BaseCSVWidget) or ( 

174 isinstance(widget, type) and 

175 issubclass(widget, BaseCSVWidget)): 

176 return widget 

177 

178 # complain since we are unable to reconstruct widget instances 

179 assert isinstance(widget, type), \ 

180 "'%s.widget' must be a widget class, not %s." \ 

181 % (self.__class__.__name__, repr(widget))

214class ChoiceIterator: 

215 # Emulates the behavior of ModelChoiceIterator, but instead wraps 

216 # the field's _choices iterable. 

217 

218 def __init__(self, field, choices): 

219 self.field = field 

220 self.choices = choices 

221 

222 def __iter__(self): 

223 if self.field.empty_label is not None: 

224 yield ("", self.field.empty_label) 

225 if self.field.null_label is not None: 

226 yield (self.field.null_value, self.field.null_label) 

227 

228 # Python 2 lacks 'yield from'
