26class Extension: 

27 """ Base class for extensions to subclass. """ 

28 

29 # Default config -- to be overriden by a subclass 

30 # Must be of the following format: 

31 # { 

32 # 'key': ['value', 'description'] 

33 # } 

34 # Note that Extension.setConfig will raise a KeyError 

35 # if a default is not set here. 

36 config = {} 

37 

38 def __init__(self, **kwargs): 

39 """ Initiate Extension and set up configs. """ 

40 self.setConfigs(kwargs) 

41 

42 def getConfig(self, key, default=''): 

43 """ Return a setting for the given key or an empty string. """ 

44 if key in self.config: 

45 return self.config[key][0] 

46 else: 

47 return default 

48 

49 def getConfigs(self): 

50 """ Return all configs settings as a dict. """ 

51 return {key: self.getConfig(key) for key in self.config.keys()} 

52 

53 def getConfigInfo(self): 

54 """ Return all config descriptions as a list of tuples. """ 

55 return [(key, self.config[key][1]) for key in self.config.keys()] 

56 

57 def setConfig(self, key, value): 

58 """ Set a config setting for `key` with the given `value`. """ 

59 if isinstance(self.config[key][0], bool): 

60 value = parseBoolValue(value) 

61 if self.config[key][0] is None: 

62 value = parseBoolValue(value, preserve_none=True) 

63 self.config[key][0] = value 

64 

65 def setConfigs(self, items): 

66 """ Set multiple config settings given a dict or list of tuples. """ 

67 if hasattr(items, 'items'): 

68 # it's a dict 

69 items = items.items() 

70 for key, value in items: 

71 self.setConfig(key, value)
