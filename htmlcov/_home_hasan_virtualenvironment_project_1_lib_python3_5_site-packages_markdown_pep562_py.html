63class Version(namedtuple("Version", ["major", "minor", "micro", "release", "pre", "post", "dev"])): 

64 """ 

65 Get the version (PEP 440). 

66 

67 A biased approach to the PEP 440 semantic version. 

68 

69 Provides a tuple structure which is sorted for comparisons `v1 > v2` etc. 

70 (major, minor, micro, release type, pre-release build, post-release build, development release build) 

71 Release types are named in is such a way they are comparable with ease. 

72 Accessors to check if a development, pre-release, or post-release build. Also provides accessor to get 

73 development status for setup files. 

74 

75 How it works (currently): 

76 

77 - You must specify a release type as either `final`, `alpha`, `beta`, or `candidate`. 

78 - To define a development release, you can use either `.dev`, `.dev-alpha`, `.dev-beta`, or `.dev-candidate`. 

79 The dot is used to ensure all development specifiers are sorted before `alpha`. 

80 You can specify a `dev` number for development builds, but do not have to as implicit development releases 

81 are allowed. 

82 - You must specify a `pre` value greater than zero if using a prerelease as this project (not PEP 440) does not 

83 allow implicit prereleases. 

84 - You can optionally set `post` to a value greater than zero to make the build a post release. While post releases 

85 are technically allowed in prereleases, it is strongly discouraged, so we are rejecting them. It should be 

86 noted that we do not allow `post0` even though PEP 440 does not restrict this. This project specifically 

87 does not allow implicit post releases. 

88 - It should be noted that we do not support epochs `1!` or local versions `+some-custom.version-1`. 

89 

90 Acceptable version releases: 

91 

92 ``` 

93 Version(1, 0, 0, "final") 1.0 

94 Version(1, 2, 0, "final") 1.2 

95 Version(1, 2, 3, "final") 1.2.3 

96 Version(1, 2, 0, ".dev-alpha", pre=4) 1.2a4 

97 Version(1, 2, 0, ".dev-beta", pre=4) 1.2b4 

98 Version(1, 2, 0, ".dev-candidate", pre=4) 1.2rc4 

99 Version(1, 2, 0, "final", post=1) 1.2.post1 

100 Version(1, 2, 3, ".dev") 1.2.3.dev0 

101 Version(1, 2, 3, ".dev", dev=1) 1.2.3.dev1 

102 ``` 

103 

104 """ 

105 

106 def __new__(cls, major, minor, micro, release="final", pre=0, post=0, dev=0): 

107 """Validate version info.""" 

108 

109 # Ensure all parts are positive integers. 

110 for value in (major, minor, micro, pre, post): 

111 if not (isinstance(value, int) and value >= 0): 

112 raise ValueError("All version parts except 'release' should be integers.") 

113 

114 if release not in REL_MAP: 

115 raise ValueError("'{}' is not a valid release type.".format(release)) 

116 

117 # Ensure valid pre-release (we do not allow implicit pre-releases). 

118 if ".dev-candidate" < release < "final": 

119 if pre == 0: 

120 raise ValueError("Implicit pre-releases not allowed.") 

121 elif dev: 

122 raise ValueError("Version is not a development release.") 

123 elif post: 

124 raise ValueError("Post-releases are not allowed with pre-releases.") 

125 

126 # Ensure valid development or development/pre release 

127 elif release < "alpha": 

128 if release > ".dev" and pre == 0: 

129 raise ValueError("Implicit pre-release not allowed.") 

130 elif post: 

131 raise ValueError("Post-releases are not allowed with pre-releases.") 

132 

133 # Ensure a valid normal release 

134 else: 

135 if pre: 

136 raise ValueError("Version is not a pre-release.") 

137 elif dev: 

138 raise ValueError("Version is not a development release.") 

139 

140 return super().__new__(cls, major, minor, micro, release, pre, post, dev) 

141 

142 def _is_pre(self): 

143 """Is prerelease.""" 

144 

145 return self.pre > 0 

146 

147 def _is_dev(self): 

148 """Is development.""" 

149 

150 return bool(self.release < "alpha") 

151 

152 def _is_post(self): 

153 """Is post.""" 

154 

155 return self.post > 0 

156 

157 def _get_dev_status(self): # pragma: no cover 

158 """Get development status string.""" 

159 

160 return DEV_STATUS[self.release] 

161 

162 def _get_canonical(self): 

163 """Get the canonical output string.""" 

164 

165 # Assemble major, minor, micro version and append `pre`, `post`, or `dev` if needed.. 

166 if self.micro == 0: 

167 ver = "{}.{}".format(self.major, self.minor) 

168 else: 

169 ver = "{}.{}.{}".format(self.major, self.minor, self.micro) 

170 if self._is_pre(): 

171 ver += '{}{}'.format(REL_MAP[self.release], self.pre) 

172 if self._is_post(): 

173 ver += ".post{}".format(self.post) 

174 if self._is_dev(): 

175 ver += ".dev{}".format(self.dev) 

176 

177 return ver

180def parse_version(ver, pre=False): 

181 """Parse version into a comparable Version tuple.""" 

182 

183 m = RE_VER.match(ver) 

184 

185 # Handle major, minor, micro 

186 major = int(m.group('major')) 

187 minor = int(m.group('minor')) if m.group('minor') else 0 

188 micro = int(m.group('micro')) if m.group('micro') else 0 

189 

190 # Handle pre releases 

191 if m.group('type'): 

192 release = PRE_REL_MAP[m.group('type')] 

193 pre = int(m.group('pre')) 

194 else: 

195 release = "final" 

196 pre = 0 

197 

198 # Handle development releases 

199 dev = m.group('dev') if m.group('dev') else 0 

200 if m.group('dev'): 

201 dev = int(m.group('dev')) 

202 release = '.dev-' + release if pre else '.dev' 

203 else: 

204 dev = 0 

205 

206 # Handle post 

207 post = int(m.group('post')) if m.group('post') else 0 

208 

209 return Version(major, minor, micro, release, pre, post, dev)

212class Pep562: 

213 """ 

214 Backport of PEP 562 <https://pypi.org/search/?q=pep562>. 

215 

216 Wraps the module in a class that exposes the mechanics to override `__dir__` and `__getattr__`. 

217 The given module will be searched for overrides of `__dir__` and `__getattr__` and use them when needed. 

218 """ 

219 

220 def __init__(self, name): 

221 """Acquire `__getattr__` and `__dir__`, but only replace module for versions less than Python 3.7.""" 

222 

223 self._module = sys.modules[name] 

224 self._get_attr = getattr(self._module, '__getattr__', None) 

225 self._get_dir = getattr(self._module, '__dir__', None) 

226 sys.modules[name] = self 

227 

228 def __dir__(self): 

229 """Return the overridden `dir` if one was provided, else apply `dir` to the module.""" 

230 

231 return self._get_dir() if self._get_dir else dir(self._module) 

232 

233 def __getattr__(self, name): 

234 """Attempt to retrieve the attribute from the module, and if missing, use the overridden function if present.""" 

235 

236 try: 

237 return getattr(self._module, name) 

238 except AttributeError: 

239 if self._get_attr: 

240 return self._get_attr(name) 

241 raise
