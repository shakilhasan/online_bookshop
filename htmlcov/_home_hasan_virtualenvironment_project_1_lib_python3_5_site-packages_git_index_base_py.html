70class IndexFile(LazyMixin, diff.Diffable, Serializable): 

71 

72 """ 

73 Implements an Index that can be manipulated using a native implementation in 

74 order to save git command function calls wherever possible. 

75 

76 It provides custom merging facilities allowing to merge without actually changing 

77 your index or your working tree. This way you can perform own test-merges based 

78 on the index only without having to deal with the working copy. This is useful 

79 in case of partial working trees. 

80 

81 ``Entries`` 

82 

83 The index contains an entries dict whose keys are tuples of type IndexEntry 

84 to facilitate access. 

85 

86 You may read the entries dict or manipulate it using IndexEntry instance, i.e.:: 

87 

88 index.entries[index.entry_key(index_entry_instance)] = index_entry_instance 

89 

90 Make sure you use index.write() once you are done manipulating the index directly 

91 before operating on it using the git command""" 

92 __slots__ = ("repo", "version", "entries", "_extension_data", "_file_path") 

93 _VERSION = 2 # latest version we support 

94 S_IFGITLINK = S_IFGITLINK # a submodule

96 def __init__(self, repo, file_path=None): 

97 """Initialize this Index instance, optionally from the given ``file_path``. 

98 If no file_path is given, we will be created from the current index file. 

99 

100 If a stream is not given, the stream will be initialized from the current 

101 repository's index on demand.""" 

102 self.repo = repo 

103 self.version = self._VERSION 

104 self._extension_data = b'' 

105 self._file_path = file_path or self._index_path()

109 # read the current index 

110 # try memory map for speed 

111 lfd = LockedFD(self._file_path)

174 def write(self, file_path=None, ignore_extension_data=False): 

175 """Write the current state to our file path or to the given one 

176 

177 :param file_path: 

178 If None, we will write to our stored file path from which we have 

179 been initialized. Otherwise we write to the given file path. 

180 Please note that this will change the file_path of this index to 

181 the one you gave. 

182 

183 :param ignore_extension_data: 

184 If True, the TREE type extension data read in the index will not 

185 be written to disk. NOTE that no extension data is actually written. 

186 Use this if you have altered the index and 

187 would like to use git-write-tree afterwards to create a tree 

188 representing your written changes. 

189 If this data is present in the written index, git-write-tree 

190 will instead write the stored/cached tree. 

191 Alternatively, use IndexFile.write_tree() to handle this case 

192 automatically 

193 

194 :return: self""" 

195 # make sure we have our entries read before getting a write lock 

196 # else it would be done when streaming. This can happen 

197 # if one doesn't change the index, but writes it right away 

198 self.entries 

199 lfd = LockedFD(file_path or self._file_path) 

200 stream = lfd.open(write=True, stream=True) 

201 

202 ok = False 

203 try: 

204 self._serialize(stream, ignore_extension_data) 

205 ok = True 

206 finally: 

207 if not ok: 

208 lfd.rollback() 

209 

210 lfd.commit() 

211 

212 # make sure we represent what we have written 

213 if file_path is not None: 

214 self._file_path = file_path

227 

228 :param base: 

229 optional treeish reference pointing to the common base of 'rhs' and

231 

232 :return:

234 conflicts ) 

235 

236 :raise GitCommandError: 

237 If there is a merge conflict. The error will 

238 be raised at the first conflicting path. If you want to have proper 

239 merge resolution to be done by yourself, you have to commit the changed

277 @classmethod 

278 def from_tree(cls, repo, *treeish, **kwargs): 

279 """Merge the given treeish revisions into a new index which is returned. 

280 The original index will remain unaltered 

281 

282 :param repo: 

283 The repository treeish are located in. 

284 

285 :param treeish: 

286 One, two or three Tree Objects, Commits or 40 byte hexshas. The result 

287 changes according to the amount of trees. 

288 If 1 Tree is given, it will just be read into a new index 

289 If 2 Trees are given, they will be merged into a new index using a 

290 two way merge algorithm. Tree 1 is the 'current' tree, tree 2 is the 'other' 

291 one. It behaves like a fast-forward. 

292 If 3 Trees are given, a 3-way merge will be performed with the first tree 

293 being the common ancestor of tree 2 and tree 3. Tree 2 is the 'current' tree, 

294 tree 3 is the 'other' one 

295 

296 :param kwargs: 

297 Additional arguments passed to git-read-tree 

298 

299 :return: 

300 New IndexFile instance. It will point to a temporary index location which 

301 does not exist anymore. If you intend to write such a merged Index, supply 

302 an alternate file_path to its 'write' method. 

303 

304 :note: 

305 In the three-way merge case, --aggressive will be specified to automatically 

306 resolve more cases in a commonly correct manner. Specify trivial=True as kwarg 

307 to override that. 

308 

309 As the underlying git-read-tree command takes into account the current index, 

310 it will be temporarily moved out of the way to assure there are no unsuspected 

311 interferences.""" 

312 if len(treeish) == 0 or len(treeish) > 3: 

313 raise ValueError("Please specify between 1 and 3 treeish, got %i" % len(treeish)) 

314 

315 arg_list = [] 

316 # ignore that working tree and index possibly are out of date 

317 if len(treeish) > 1: 

318 # drop unmerged entries when reading our index and merging 

319 arg_list.append("--reset") 

320 # handle non-trivial cases the way a real merge does 

321 arg_list.append("--aggressive") 

322 # END merge handling 

323 

324 # tmp file created in git home directory to be sure renaming 

325 # works - /tmp/ dirs could be on another device 

326 tmp_index = tempfile.mktemp('', '', repo.git_dir) 

327 arg_list.append("--index-output=%s" % tmp_index) 

328 arg_list.extend(treeish) 

329 

330 # move current index out of the way - otherwise the merge may fail 

331 # as it considers existing entries. moving it essentially clears the index. 

332 # Unfortunately there is no 'soft' way to do it. 

333 # The TemporaryFileSwap assure the original file get put back 

334 index_handler = TemporaryFileSwap(join_path_native(repo.git_dir, 'index')) 

335 try: 

336 repo.git.read_tree(*arg_list, **kwargs) 

337 index = cls(repo, tmp_index) 

338 index.entries # force it to read the file as we will delete the temp-file 

339 del(index_handler) # release as soon as possible 

340 finally: 

341 if osp.exists(tmp_index): 

342 os.remove(tmp_index) 

343 # END index merge handling 

344 

345 return index

349 def _iter_expand_paths(self, paths): 

350 """Expand the directories in list of paths to the corresponding paths accordingly, 

351 

352 Note: git will add items multiple times even if a glob overlapped 

353 with manually specified paths or if paths where specified multiple 

354 times - we respect that and do not prune""" 

355 def raise_exc(e): 

356 raise e 

357 r = self.repo.working_tree_dir 

358 rs = r + os.sep 

359 for path in paths: 

360 abs_path = path 

361 if not osp.isabs(abs_path): 

362 abs_path = osp.join(r, path) 

363 # END make absolute path 

364 

365 try: 

366 st = os.lstat(abs_path) # handles non-symlinks as well 

367 except OSError: 

368 # the lstat call may fail as the path may contain globs as well 

369 pass 

370 else: 

371 if S_ISLNK(st.st_mode): 

372 yield abs_path.replace(rs, '') 

373 continue 

374 # end check symlink 

375 

376 # if the path is not already pointing to an existing file, resolve globs if possible 

377 if not os.path.exists(path) and ('?' in path or '*' in path or '[' in path): 

378 resolved_paths = glob.glob(abs_path) 

379 # not abs_path in resolved_paths: 

380 # a glob() resolving to the same path we are feeding it with 

381 # is a glob() that failed to resolve. If we continued calling 

382 # ourselves we'd endlessly recurse. If the condition below 

383 # evaluates to true then we are likely dealing with a file 

384 # whose name contains wildcard characters. 

385 if abs_path not in resolved_paths: 

386 for f in self._iter_expand_paths(glob.glob(abs_path)): 

387 yield f.replace(rs, '') 

388 continue 

389 # END glob handling 

390 try: 

391 for root, _dirs, files in os.walk(abs_path, onerror=raise_exc): 

392 for rela_file in files: 

393 # add relative paths only 

394 yield osp.join(root.replace(rs, ''), rela_file) 

395 # END for each file in subdir 

396 # END for each subdirectory 

397 except OSError: 

398 # was a file or something that could not be iterated 

399 yield path.replace(rs, '') 

400 # END path exception handling 

401 # END for each path

403 def _write_path_to_stdin(self, proc, filepath, item, fmakeexc, fprogress, 

404 read_from_stdout=True): 

405 """Write path to proc.stdin and make sure it processes the item, including progress. 

406 

407 :return: stdout string 

408 :param read_from_stdout: if True, proc.stdout will be read after the item 

409 was sent to stdin. In that case, it will return None 

410 :note: There is a bug in git-update-index that prevents it from sending 

411 reports just in time. This is why we have a version that tries to 

412 read stdout and one which doesn't. In fact, the stdout is not 

413 important as the piped-in files are processed anyway and just in time 

414 :note: Newlines are essential here, gits behaviour is somewhat inconsistent 

415 on this depending on the version, hence we try our best to deal with 

416 newlines carefully. Usually the last newline will not be sent, instead 

417 we will close stdin to break the pipe.""" 

418 

419 fprogress(filepath, False, item) 

420 rval = None 

421 try: 

422 proc.stdin.write(("%s\n" % filepath).encode(defenc)) 

423 except IOError as e: 

424 # pipe broke, usually because some error happened 

425 raise fmakeexc() from e 

426 # END write exception handling 

427 proc.stdin.flush() 

428 if read_from_stdout: 

429 rval = proc.stdout.readline().strip() 

430 fprogress(filepath, True, item) 

431 return rval

449 def unmerged_blobs(self): 

450 """ 

451 :return: 

452 Iterator yielding dict(path : list( tuple( stage, Blob, ...))), being 

453 a dictionary associating a path in the index with a list containing 

454 sorted stage/blob pairs 

455 

456 :note: 

457 Blobs that have been removed in one side simply do not exist in the 

458 given stage. I.e. a file removed on the 'other' branch whose entries 

459 are at stage 3 will not have a stage 3 entry. 

460 """ 

461 is_unmerged_blob = lambda t: t[0] != 0 

462 path_map = {} 

463 for stage, blob in self.iter_blobs(is_unmerged_blob): 

464 path_map.setdefault(blob.path, []).append((stage, blob)) 

465 # END for each unmerged blob 

466 for line in path_map.values(): 

467 line.sort() 

468 return path_map

474 def resolve_blobs(self, iter_blobs): 

475 """Resolve the blobs given in blob iterator. This will effectively remove the 

476 index entries of the respective path at all non-null stages and add the given 

477 blob as new stage null blob. 

478 

479 For each path there may only be one blob, otherwise a ValueError will be raised 

480 claiming the path is already at stage 0. 

481 

482 :raise ValueError: if one of the blobs already existed at stage 0 

483 :return: self 

484 

485 :note: 

486 You will have to write the index manually once you are done, i.e. 

487 index.resolve_blobs(blobs).write() 

488 """ 

489 for blob in iter_blobs: 

490 stage_null_key = (blob.path, 0) 

491 if stage_null_key in self.entries: 

492 raise ValueError("Path %r already exists at stage 0" % blob.path) 

493 # END assert blob is not stage 0 already 

494 

495 # delete all possible stages 

496 for stage in (1, 2, 3): 

497 try: 

498 del(self.entries[(blob.path, stage)]) 

499 except KeyError: 

500 pass 

501 # END ignore key errors 

502 # END for each possible stage 

503 

504 self.entries[stage_null_key] = IndexEntry.from_blob(blob) 

505 # END for each blob 

506 

507 return self

597 return BaseIndexEntry((stat_mode_to_index_mode(st.st_mode), 

598 istream.binsha, 0, to_native_path_linux(filepath)))

631 def add(self, items, force=True, fprogress=lambda *args: None, path_rewriter=None, 

632 write=True, write_extension_data=False): 

633 """Add files from the working tree, specific blobs or BaseIndexEntries 

634 to the index. 

635 

636 :param items: 

637 Multiple types of items are supported, types can be mixed within one call. 

638 Different types imply a different handling. File paths may generally be 

639 relative or absolute. 

640 

641 - path string 

642 strings denote a relative or absolute path into the repository pointing to 

643 an existing file, i.e. CHANGES, lib/myfile.ext, '/home/gitrepo/lib/myfile.ext'. 

644 

645 Absolute paths must start with working tree directory of this index's repository 

646 to be considered valid. For example, if it was initialized with a non-normalized path, like 

647 `/root/repo/../repo`, absolute paths to be added must start with `/root/repo/../repo`. 

648 

649 Paths provided like this must exist. When added, they will be written 

650 into the object database. 

651 

652 PathStrings may contain globs, such as 'lib/__init__*' or can be directories 

653 like 'lib', the latter ones will add all the files within the dirctory and 

654 subdirectories. 

655 

656 This equals a straight git-add. 

657
