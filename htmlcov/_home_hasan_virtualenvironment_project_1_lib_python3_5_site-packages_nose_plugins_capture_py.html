23class Capture(Plugin): 

24 """ 

25 Output capture plugin. Enabled by default. Disable with ``-s`` or 

26 ``--nocapture``. This plugin captures stdout during test execution, 

27 appending any output captured to the error or failure output, 

28 should the test fail or raise an error. 

29 """ 

30 enabled = True 

31 env_opt = 'NOSE_NOCAPTURE' 

32 name = 'capture'

58 self.end()

71 def formatError(self, test, err): 

72 """Add captured output to error report. 

73 """ 

74 test.capturedOutput = output = self.buffer 

75 self._buf = None 

76 if not output: 

77 # Don't return None as that will prevent other 

78 # formatters from formatting and remove earlier formatters 

79 # formats, instead return the err we got 

80 return err 

81 ec, ev, tb = err 

82 return (ec, self.addCaptureToErr(ev, output), tb) 

83 

84 def formatFailure(self, test, err): 

85 """Add captured output to failure report. 

86 """ 

87 return self.formatError(test, err)

89 def addCaptureToErr(self, ev, output): 

90 ev = exc_to_unicode(ev) 

91 output = force_unicode(output) 

92 return '\n'.join([ev, ln('>> begin captured stdout <<'), 

93 output, ln('>> end captured stdout <<')])

95 def start(self): 

96 self.stdout.append(sys.stdout)

99 

100 def end(self): 

101 if self.stdout: 

102 sys.stdout = self.stdout.pop() 

103 

104 def finalize(self, result): 

105 """Restore stdout. 

106 """ 

107 while self.stdout: 

108 self.end()
