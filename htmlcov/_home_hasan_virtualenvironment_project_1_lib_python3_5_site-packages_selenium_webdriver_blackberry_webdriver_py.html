34class WebDriver(RemoteWebDriver): 

35 """ 

36 Controls the BlackBerry Browser and allows you to drive it. 

37 

38 :Args: 

39 - device_password - password for the BlackBerry device or emulator you are 

40 trying to drive 

41 - bb_tools_dir path to the blackberry-deploy executable. If the default 

42 is used it assumes it is in the $PATH 

43 - hostip - the ip for the device you are trying to drive. Falls back to 

44 169.254.0.1 which is the default ip used 

45 - port - the port being used for WebDriver on device. defaults to 1338 

46 - desired_capabilities: Dictionary object with non-browser specific 

47 capabilities only, such as "proxy" or "loggingPref". 

48 

49 Note: To get blackberry-deploy you will need to install the BlackBerry 

50 WebWorks SDK - the default install will put it in the $PATH for you. 

51 Download at https://developer.blackberry.com/html5/downloads/ 

52 """ 

53 def __init__(self, device_password, bb_tools_dir=None, 

54 hostip='169.254.0.1', port=1338, desired_capabilities={}): 

55 remote_addr = 'http://{}:{}'.format(hostip, port) 

56 

57 filename = 'blackberry-deploy' 

58 if platform.system() == "Windows": 

59 filename += '.bat' 

60 

61 if bb_tools_dir is not None: 

62 if os.path.isdir(bb_tools_dir): 

63 bb_deploy_location = os.path.join(bb_tools_dir, filename) 

64 if not os.path.isfile(bb_deploy_location): 

65 raise WebDriverException('Invalid blackberry-deploy location: {}'.format(bb_deploy_location)) 

66 else: 

67 raise WebDriverException('Invalid blackberry tools location, must be a directory: {}'.format(bb_tools_dir)) 

68 else: 

69 bb_deploy_location = filename 

70 

71 """ 

72 Now launch the BlackBerry browser before allowing anything else to run. 

73 """ 

74 try: 

75 launch_args = [bb_deploy_location, 

76 '-launchApp', 

77 str(hostip), 

78 '-package-name', 'sys.browser', 

79 '-package-id', 'gYABgJYFHAzbeFMPCCpYWBtHAm0', 

80 '-password', str(device_password)] 

81 

82 with open(os.devnull, 'w') as fp: 

83 p = subprocess.Popen(launch_args, stdout=fp) 

84 

85 returncode = p.wait() 

86 

87 if returncode == 0: 

88 # wait for the BlackBerry10 browser to load. 

89 is_running_args = [bb_deploy_location, 

90 '-isAppRunning', 

91 str(hostip), 

92 '-package-name', 'sys.browser', 

93 '-package-id', 'gYABgJYFHAzbeFMPCCpYWBtHAm0', 

94 '-password', str(device_password)] 

95 

96 WebDriverWait(None, LOAD_TIMEOUT)\ 

97 .until(lambda x: subprocess.check_output(is_running_args) 

98 .find('result::true'), 

99 message='waiting for BlackBerry10 browser to load') 

100 

101 RemoteWebDriver.__init__(self, 

102 command_executor=remote_addr, 

103 desired_capabilities=desired_capabilities) 

104 else: 

105 raise WebDriverException('blackberry-deploy failed to launch browser') 

106 except Exception as e: 

107 raise WebDriverException('Something went wrong launching blackberry-deploy', stacktrace=getattr(e, 'stacktrace', None))
