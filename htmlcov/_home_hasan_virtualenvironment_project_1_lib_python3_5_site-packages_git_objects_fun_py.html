12def tree_to_stream(entries, write): 

13 """Write the give list of entries into a stream using its write method 

14 :param entries: **sorted** list of tuples with (binsha, mode, name) 

15 :param write: write method which takes a data string""" 

16 ord_zero = ord('0') 

17 bit_mask = 7 # 3 bits set 

18 

19 for binsha, mode, name in entries: 

20 mode_str = b'' 

21 for i in range(6): 

22 mode_str = bytes([((mode >> (i * 3)) & bit_mask) + ord_zero]) + mode_str 

23 # END for each 8 octal value 

24 

25 # git slices away the first octal if its zero 

26 if mode_str[0] == ord_zero: 

27 mode_str = mode_str[1:] 

28 # END save a byte 

29 

30 # here it comes: if the name is actually unicode, the replacement below 

31 # will not work as the binsha is not part of the ascii unicode encoding - 

32 # hence we must convert to an utf8 string for it to work properly. 

33 # According to my tests, this is exactly what git does, that is it just 

34 # takes the input literally, which appears to be utf8 on linux. 

35 if isinstance(name, str): 

36 name = name.encode(defenc) 

37 write(b''.join((mode_str, b' ', name, b'\0', binsha))) 

38 # END for each item

41def tree_entries_from_data(data): 

42 """Reads the binary representation of a tree and returns tuples of Tree items 

43 :param data: data block with tree data (as bytes) 

44 :return: list(tuple(binsha, mode, tree_relative_path), ...)""" 

45 ord_zero = ord('0') 

46 space_ord = ord(' ') 

47 len_data = len(data) 

48 i = 0 

49 out = [] 

50 while i < len_data: 

51 mode = 0 

52 

53 # read mode 

54 # Some git versions truncate the leading 0, some don't 

55 # The type will be extracted from the mode later 

56 while data[i] != space_ord: 

57 # move existing mode integer up one level being 3 bits 

58 # and add the actual ordinal value of the character 

59 mode = (mode << 3) + (data[i] - ord_zero) 

60 i += 1 

61 # END while reading mode 

62 

63 # byte is space now, skip it 

64 i += 1 

65 

66 # parse name, it is NULL separated 

67 

68 ns = i 

69 while data[i] != 0: 

70 i += 1 

71 # END while not reached NULL 

72 

73 # default encoding for strings in git is utf8 

74 # Only use the respective unicode object if the byte stream was encoded 

75 name = data[ns:i] 

76 name = safe_decode(name) 

77 

78 # byte is NULL, get next 20 

79 i += 1 

80 sha = data[i:i + 20] 

81 i = i + 20 

82 out.append((sha, mode, name)) 

83 # END for each byte in data stream 

84 return out

87def _find_by_name(tree_data, name, is_dir, start_at): 

88 """return data entry matching the given name and tree mode 

89 or None. 

90 Before the item is returned, the respective data item is set 

91 None in the tree_data list to mark it done""" 

92 try: 

93 item = tree_data[start_at] 

94 if item and item[2] == name and S_ISDIR(item[1]) == is_dir: 

95 tree_data[start_at] = None 

96 return item 

97 except IndexError: 

98 pass 

99 # END exception handling 

100 for index, item in enumerate(tree_data): 

101 if item and item[2] == name and S_ISDIR(item[1]) == is_dir: 

102 tree_data[index] = None 

103 return item 

104 # END if item matches 

105 # END for each item 

106 return None

116def traverse_trees_recursive(odb, tree_shas, path_prefix): 

117 """ 

118 :return: list with entries according to the given binary tree-shas. 

119 The result is encoded in a list 

120 of n tuple|None per blob/commit, (n == len(tree_shas)), where 

121 * [0] == 20 byte sha 

122 * [1] == mode as int 

123 * [2] == path relative to working tree root 

124 The entry tuple is None if the respective blob/commit did not 

125 exist in the given tree. 

126 :param tree_shas: iterable of shas pointing to trees. All trees must 

127 be on the same level. A tree-sha may be None in which case None 

128 :param path_prefix: a prefix to be added to the returned paths on this level, 

129 set it '' for the first iteration 

130 :note: The ordering of the returned items will be partially lost""" 

131 trees_data = [] 

132 nt = len(tree_shas) 

133 for tree_sha in tree_shas: 

134 if tree_sha is None: 

135 data = [] 

136 else: 

137 data = tree_entries_from_data(odb.stream(tree_sha).read()) 

138 # END handle muted trees 

139 trees_data.append(data) 

140 # END for each sha to get data for 

141 

142 out = [] 

143 out_append = out.append 

144 

145 # find all matching entries and recursively process them together if the match 

146 # is a tree. If the match is a non-tree item, put it into the result. 

147 # Processed items will be set None 

148 for ti, tree_data in enumerate(trees_data): 

149 for ii, item in enumerate(tree_data): 

150 if not item: 

151 continue 

152 # END skip already done items 

153 entries = [None for _ in range(nt)] 

154 entries[ti] = item 

155 _sha, mode, name = item 

156 is_dir = S_ISDIR(mode) # type mode bits 

157 

158 # find this item in all other tree data items 

159 # wrap around, but stop one before our current index, hence 

160 # ti+nt, not ti+1+nt 

161 for tio in range(ti + 1, ti + nt): 

162 tio = tio % nt 

163 entries[tio] = _find_by_name(trees_data[tio], name, is_dir, ii) 

164 # END for each other item data 

165 

166 # if we are a directory, enter recursion 

167 if is_dir: 

168 out.extend(traverse_trees_recursive( 

169 odb, [((ei and ei[0]) or None) for ei in entries], path_prefix + name + '/')) 

170 else: 

171 out_append(tuple(_to_full_path(e, path_prefix) for e in entries)) 

172 # END handle recursion 

173 

174 # finally mark it done 

175 tree_data[ii] = None 

176 # END for each item 

177 

178 # we are done with one tree, set all its data empty 

179 del(tree_data[:]) 

180 # END for each tree_data chunk 

181 return out

184def traverse_tree_recursive(odb, tree_sha, path_prefix): 

185 """ 

186 :return: list of entries of the tree pointed to by the binary tree_sha. An entry 

187 has the following format: 

188 * [0] 20 byte sha 

189 * [1] mode as int 

190 * [2] path relative to the repository 

191 :param path_prefix: prefix to prepend to the front of all returned paths""" 

192 entries = [] 

193 data = tree_entries_from_data(odb.stream(tree_sha).read()) 

194 

195 # unpacking/packing is faster than accessing individual items 

196 for sha, mode, name in data: 

197 if S_ISDIR(mode): 

198 entries.extend(traverse_tree_recursive(odb, sha, path_prefix + name + '/')) 

199 else: 

200 entries.append((sha, mode, path_prefix + name)) 

201 # END for each item 

202 

203 return entries
