116 def run(self, result): 

117 """Modified run for the test wrapper. 

118 

119 From here we don't call result.startTest or stopTest or 

120 addSuccess. The wrapper calls addError/addFailure only if its 

121 own setup or teardown fails, or running the wrapped test fails 

122 (eg, if the wrapped "test" is not callable). 

123 

124 Two additional methods are called, beforeTest and 

125 afterTest. These give plugins a chance to modify the wrapped 

126 test before it is called and do cleanup after it is 

127 called. They are called unconditionally. 

128 """ 

129 if self.resultProxy: 

130 result = self.resultProxy(result, self) 

131 try: 

132 try: 

133 self.beforeTest(result) 

134 self.runTest(result) 

135 except KeyboardInterrupt: 

136 raise 

137 except: 

138 err = sys.exc_info() 

139 result.addError(self, err) 

140 finally: 

141 self.afterTest(result)

143 def runTest(self, result): 

144 """Run the test. Plugins may alter the test by returning a 

145 value from prepareTestCase. The value must be callable and 

146 must accept one argument, the result instance. 

147 """ 

148 test = self.test 

149 plug_test = self.config.plugins.prepareTestCase(self) 

150 if plug_test is not None: 

151 test = plug_test 

152 test(result)

308class MethodTestCase(TestBase): 

309 """Test case wrapper for test methods. 

310 

311 Don't use this class directly; it is used internally in nose to 

312 create test cases for test methods. 

313 """ 

314 __test__ = False # do not collect 

315 

316 def __init__(self, method, test=None, arg=tuple(), descriptor=None): 

317 """Initialize the MethodTestCase. 

318 

319 Required argument: 

320 

321 * method -- the method to call, may be bound or unbound. In either 

322 case, a new instance of the method's class will be instantiated to 

323 make the call. Note: In Python 3.x, if using an unbound method, you 

324 must wrap it using pyversion.unbound_method. 

325 

326 Optional arguments: 

327 

328 * test -- the test function to call. If this is passed, it will be 

329 called instead of getting a new bound method of the same name as the 

330 desired method from the test instance. This is to support generator 

331 methods that yield inline functions. 

332 

333 * arg -- arguments to pass to the test function. This is to support 

334 generator methods that yield arguments. 

335 

336 * descriptor -- the function, other than the test, that should be used 

337 to construct the test name. This is to support generator methods. 

338 """ 

339 self.method = method 

340 self.test = test 

341 self.arg = arg 

342 self.descriptor = descriptor 

343 if isfunction(method): 

344 raise ValueError("Unbound methods must be wrapped using pyversion.unbound_method before passing to MethodTestCase") 

345 self.cls = method.__self__.__class__ 

346 self.inst = self.cls() 

347 if self.test is None: 

348 method_name = self.method.__name__ 

349 self.test = getattr(self.inst, method_name) 

350 TestBase.__init__(self) 

351 

352 def __str__(self): 

353 func, arg = self._descriptors() 

354 if hasattr(func, 'compat_func_name'): 

355 name = func.compat_func_name 

356 else: 

357 name = func.__name__ 

358 name = "%s.%s.%s" % (self.cls.__module__, 

359 self.cls.__name__, 

360 name) 

361 if arg: 

362 name = "%s%s" % (name, arg) 

363 return name
