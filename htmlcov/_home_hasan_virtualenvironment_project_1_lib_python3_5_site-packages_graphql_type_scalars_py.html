10# As per the GraphQL Spec, Integers are only treated as valid when a valid 

11# 32-bit signed integer, providing the broadest support across platforms. 

12# 

13# n.b. JavaScript's integers are safe between -(2^31 - 1) and 2^31 - 1 because 

14# they are internally represented as IEEE 754 doubles. 

15MAX_INT = 2147483647 

16MIN_INT = -2147483648

19def coerce_int(value): 

20 # type: (Any) -> int 

21 if isinstance(value, int): 

22 num = value 

23 else: 

24 try: 

25 num = int(value) 

26 except ValueError: 

27 num = int(float(value)) 

28 if MIN_INT <= num <= MAX_INT: 

29 return num 

30 

31 raise Exception( 

32 ("Int cannot represent non 32-bit signed integer value: {}").format(value) 

33 )

36def parse_int_literal(ast): 

37 # type: (IntValue) -> Optional[int] 

38 if isinstance(ast, IntValue): 

39 num = int(ast.value) 

40 if MIN_INT <= num <= MAX_INT: 

41 return num 

42 return None
