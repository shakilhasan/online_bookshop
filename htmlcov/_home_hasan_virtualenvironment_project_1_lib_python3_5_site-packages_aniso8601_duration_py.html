16def parse_duration(isodurationstr, builder=PythonTimeBuilder): 

17 #Given a string representing an ISO 8601 duration, return a 

18 #a duration built by the given builder. Valid formats are: 

19 # 

20 #PnYnMnDTnHnMnS (or any reduced precision equivalent) 

21 #P<date>T<time> 

22 

23 if isodurationstr[0] != 'P': 

24 raise ISOFormatError('ISO 8601 duration must start with a P.') 

25 

26 #If Y, M, D, H, S, or W are in the string, 

27 #assume it is a specified duration 

28 if _has_any_component(isodurationstr, 

29 ['Y', 'M', 'D', 'H', 'S', 'W']) is True: 

30 return _parse_duration_prescribed(isodurationstr, builder) 

31 

32 return _parse_duration_combined(isodurationstr, builder)

34def _parse_duration_prescribed(durationstr, builder): 

35 #durationstr can be of the form PnYnMnDTnHnMnS or PnW 

36 

37 #Don't allow negative elements 

38 #https://bitbucket.org/nielsenb/aniso8601/issues/20/negative-duration 

39 if durationstr.find('-') != -1: 

40 raise NegativeDurationError('ISO 8601 durations must be positive.') 

41 

42 #Make sure the end character is valid 

43 #https://bitbucket.org/nielsenb/aniso8601/issues/9/durations-with-trailing-garbage-are-parsed 

44 if durationstr[-1] not in ['Y', 'M', 'D', 'H', 'S', 'W']: 

45 raise ISOFormatError('ISO 8601 duration must end with a valid ' 

46 'character.') 

47 

48 #Make sure only the lowest order element has decimal precision 

49 if durationstr.count('.') > 1: 

50 raise ISOFormatError('ISO 8601 allows only lowest order element to ' 

51 'have a decimal fraction.') 

52 elif durationstr.count('.') == 1: 

53 #There should only ever be 1 letter after a decimal if there is more 

54 #then one, the string is invalid 

55 lettercount = 0 

56 

57 for character in durationstr.split('.')[1]: 

58 if character.isalpha() is True: 

59 lettercount += 1 

60 

61 if lettercount > 1: 

62 raise ISOFormatError('ISO 8601 duration must end with ' 

63 'a single valid character.') 

64 

65 #Do not allow W in combination with other designators 

66 #https://bitbucket.org/nielsenb/aniso8601/issues/2/week-designators-should-not-be-combinable 

67 if (durationstr.find('W') != -1 

68 and _has_any_component(durationstr, 

69 ['Y', 'M', 'D', 'H', 'S']) is True): 

70 raise ISOFormatError('ISO 8601 week designators may not be combined ' 

71 'with other time designators.') 

72 

73 #Parse the elements of the duration 

74 if durationstr.find('T') == -1: 

75 return _parse_duration_prescribed_notime(durationstr, builder) 

76 

77 return _parse_duration_prescribed_time(durationstr, builder)

79def _parse_duration_prescribed_notime(durationstr, builder): 

80 #durationstr can be of the form PnYnMnD or PnW 

81 

82 #Don't allow negative elements 

83 #https://bitbucket.org/nielsenb/aniso8601/issues/20/negative-duration 

84 if durationstr.find('-') != -1: 

85 raise NegativeDurationError('ISO 8601 durations must be positive.') 

86 

87 #Make sure no time portion is included 

88 #https://bitbucket.org/nielsenb/aniso8601/issues/7/durations-with-time-components-before-t 

89 if _has_any_component(durationstr, ['H', 'S']): 

90 raise ISOFormatError('ISO 8601 time components not allowed in duration ' 

91 'without prescribed time.') 

92 

93 if _component_order_correct(durationstr, 

94 ['P', 'Y', 'M', 'D', 'W']) is False: 

95 raise ISOFormatError('ISO 8601 duration components must be in the ' 

96 'correct order.') 

97 

98 if durationstr.find('Y') != -1: 

99 yearstr = _parse_duration_element(durationstr, 'Y') 

100 else: 

101 yearstr = None 

102 

103 if durationstr.find('M') != -1: 

104 monthstr = _parse_duration_element(durationstr, 'M') 

105 else: 

106 monthstr = None 

107 

108 if durationstr.find('W') != -1: 

109 weekstr = _parse_duration_element(durationstr, 'W') 

110 else: 

111 weekstr = None 

112 

113 if durationstr.find('D') != -1: 

114 daystr = _parse_duration_element(durationstr, 'D') 

115 else: 

116 daystr = None 

117 

118 return builder.build_duration(PnY=yearstr, PnM=monthstr, 

119 PnW=weekstr, PnD=daystr)

121def _parse_duration_prescribed_time(durationstr, builder): 

122 #durationstr can be of the form PnYnMnDTnHnMnS 

123 

124 #Don't allow negative elements 

125 #https://bitbucket.org/nielsenb/aniso8601/issues/20/negative-duration 

126 if durationstr.find('-') != -1: 

127 raise NegativeDurationError('ISO 8601 durations must be positive.') 

128 

129 firsthalf = durationstr[:durationstr.find('T')] 

130 secondhalf = durationstr[durationstr.find('T'):] 

131 

132 #Make sure no time portion is included in the date half 

133 #https://bitbucket.org/nielsenb/aniso8601/issues/7/durations-with-time-components-before-t 

134 if _has_any_component(firsthalf, ['H', 'S']): 

135 raise ISOFormatError('ISO 8601 time components not allowed in date ' 

136 'portion of duration.') 

137 

138 if _component_order_correct(firsthalf, ['P', 'Y', 'M', 'D', 'W']) is False: 

139 raise ISOFormatError('ISO 8601 duration components must be in the ' 

140 'correct order.') 

141 

142 #Make sure no date component is included in the time half 

143 if _has_any_component(secondhalf, ['Y', 'D']): 

144 raise ISOFormatError('ISO 8601 time components not allowed in date ' 

145 'portion of duration.')

178 else:

184def _parse_duration_combined(durationstr, builder): 

185 #Period of the form P<date>T<time> 

186 

187 #Split the string in to its component parts 

188 datepart, timepart = durationstr[1:].split('T') #We skip the 'P' 

189 

190 datevalue = parse_date(datepart, builder=TupleBuilder) 

191 timevalue = parse_time(timepart, builder=TupleBuilder) 

192 

193 return builder.build_duration(PnY=datevalue[0], PnM=datevalue[1], 

194 PnD=datevalue[2], TnH=timevalue[0], 

195 TnM=timevalue[1], TnS=timevalue[2])

197def _parse_duration_element(durationstr, elementstr): 

198 #Extracts the specified portion of a duration, for instance, given: 

199 #durationstr = 'T4H5M6.1234S' 

200 #elementstr = 'H' 

201 # 

202 #returns 4 

203 # 

204 #Note that the string must start with a character, so its assumed the 

205 #full duration string would be split at the 'T' 

206 

207 durationstartindex = 0 

208 durationendindex = durationstr.find(elementstr) 

209 

210 for characterindex in compat.range(durationendindex - 1, 0, -1): 

211 if durationstr[characterindex].isalpha() is True: 

212 durationstartindex = characterindex 

213 break 

214 

215 durationstartindex += 1 

216 

217 if ',' in durationstr: 

218 #Replace the comma with a 'full-stop' 

219 durationstr = durationstr.replace(',', '.') 

220 

221 return durationstr[durationstartindex:durationendindex]

223def _has_any_component(durationstr, components): 

224 #Given a duration string, and a list of components, returns True 

225 #if any of the listed components are present, False otherwise. 

226 # 

227 #For instance: 

228 #durationstr = 'P1Y' 

229 #components = ['Y', 'M'] 

230 # 

231 #returns True 

232 # 

233 #durationstr = 'P1Y' 

234 #components = ['M', 'D'] 

235 # 

236 #returns False 

237 

238 for component in components: 

239 if durationstr.find(component) != -1: 

240 return True 

241 

242 return False

244def _component_order_correct(durationstr, componentorder): 

245 #Given a duration string, and a list of components, returns 

246 #True if the components are in the same order as the 

247 #component order list, False otherwise. Characters that 

248 #are present in the component order list but not in the 

249 #duration string are ignored. 

250 # 

251 #https://bitbucket.org/nielsenb/aniso8601/issues/8/durations-with-components-in-wrong-order 

252 # 

253 #durationstr = 'P1Y1M1D' 

254 #components = ['P', 'Y', 'M', 'D'] 

255 # 

256 #returns True 

257 # 

258 #durationstr = 'P1Y1M' 

259 #components = ['P', 'Y', 'M', 'D'] 

260 # 

261 #returns True 

262 # 

263 #durationstr = 'P1D1Y1M' 

264 #components = ['P', 'Y', 'M', 'D'] 

265 # 

266 #returns False 

267 

268 componentindex = 0 

269 

270 for characterindex in compat.range(len(durationstr)): 

271 character = durationstr[characterindex] 

272 

273 if character in componentorder: 

274 #This is a character we need to check the order of 

275 if character in componentorder[componentindex:]: 

276 componentindex = componentorder.index(character) 

277 else: 

278 #A character is out of order 

279 return False 

280 

281 return True
