88 def __init__(self, path=None, odbt=GitCmdObjectDB, search_parent_directories=False, expand_vars=True): 

89 """Create a new Repo instance 

90 

91 :param path: 

92 the path to either the root git directory or the bare git repo:: 

93 

94 repo = Repo("/Users/mtrier/Development/git-python") 

95 repo = Repo("/Users/mtrier/Development/git-python.git") 

96 repo = Repo("~/Development/git-python.git") 

97 repo = Repo("$REPOSITORIES/Development/git-python.git") 

98 repo = Repo("C:\\Users\\mtrier\\Development\\git-python\\.git") 

99 

100 - In *Cygwin*, path may be a `'cygdrive/...'` prefixed path. 

101 - If it evaluates to false, :envvar:`GIT_DIR` is used, and if this also evals to false, 

102 the current-directory is used. 

103 :param odbt: 

104 Object DataBase type - a type which is constructed by providing 

105 the directory containing the database objects, i.e. .git/objects. It will 

106 be used to access all object data 

107 :param search_parent_directories: 

108 if True, all parent directories will be searched for a valid repo as well. 

109 

110 Please note that this was the default behaviour in older versions of GitPython, 

111 which is considered a bug though. 

112 :raise InvalidGitRepositoryError: 

113 :raise NoSuchPathError: 

114 :return: git.Repo """

129 if not os.path.exists(epath):

132 ## Walk up the path to find the `.git` dir. 

133 #

136 # ABOUT osp.NORMPATH 

137 # It's important to normalize the paths, as submodules will otherwise initialize their 

138 # repo instances with paths that depend on path-portions that will not exist after being 

139 # removed. It's just cleaner.

144 # variable is set, core.worktree is ignored and not used for determining the 

145 # root of working tree. This can be overridden by the GIT_WORK_TREE environment 

146 # variable. The value can be an absolute path or relative to the path to the .git 

147 # directory, which is either specified by GIT_DIR, or automatically discovered. 

148 # If GIT_DIR is specified but none of GIT_WORK_TREE and core.worktree is specified, 

149 # the current working directory is regarded as the top level of your working tree.

152 gitconf = self.config_reader("repository") 

153 if gitconf.has_option('core', 'worktree'):

155 if 'GIT_WORK_TREE' in os.environ: 

156 self._working_tree_dir = os.getenv('GIT_WORK_TREE') 

157 break 

158 

159 dotgit = osp.join(curpath, '.git') 

160 sm_gitpath = find_submodule_git_dir(dotgit) 

161 if sm_gitpath is not None: 

162 self.git_dir = osp.normpath(sm_gitpath) 

163 

164 sm_gitpath = find_submodule_git_dir(dotgit) 

165 if sm_gitpath is None: 

166 sm_gitpath = find_worktree_git_dir(dotgit) 

167 

168 if sm_gitpath is not None: 

169 self.git_dir = expand_path(sm_gitpath, expand_vars) 

170 self._working_tree_dir = curpath 

171 break 

172 

173 if not search_parent_directories: 

174 break 

175 curpath, tail = osp.split(curpath) 

176 if not tail: 

177 break 

178 # END while curpath 

179 

180 if self.git_dir is None: 

181 raise InvalidGitRepositoryError(epath) 

182 

183 self._bare = False 

184 try: 

185 self._bare = self.config_reader("repository").getboolean('core', 'bare') 

186 except Exception:

188 pass

191 common_dir = open(osp.join(self.git_dir, 'commondir'), 'rt').readlines()[0].strip()

193 except (OSError, IOError): 

194 self._common_dir = None 

195

197 # in the first place 

198 if self._bare: 

199 self._working_tree_dir = None 

200 # END working dir handling

203 self.git = self.GitCommandWrapperType(self.working_dir)

528 def merge_base(self, *rev, **kwargs): 

529 """Find the closest common ancestor for the given revision (e.g. Commits, Tags, References, etc) 

530 

531 :param rev: At least two revs to find the common ancestor for. 

532 :param kwargs: Additional arguments to be passed to the repo.git.merge_base() command which does all the work. 

533 :return: A list of Commit objects. If --all was not specified as kwarg, the list will have at max one Commit, 

534 or is empty if no common merge base exists. 

535 :raises ValueError: If not at least two revs are provided 

536 """ 

537 if len(rev) < 2: 

538 raise ValueError("Please specify at least two revs, got only %i" % len(rev)) 

539 # end handle input 

540 

541 res = [] 

542 try: 

543 lines = self.git.merge_base(*rev, **kwargs).splitlines() 

544 except GitCommandError as err: 

545 if err.status == 128: 

546 raise 

547 # end handle invalid rev 

548 # Status code 1 is returned if there is no merge-base 

549 # (see https://github.com/git/git/blob/master/builtin/merge-base.c#L16) 

550 return res 

551 # end exception handling 

552 

553 for line in lines: 

554 res.append(self.commit(line)) 

555 # end for each merge-base 

556 

557 return res

625 def is_dirty(self, index=True, working_tree=True, untracked_files=False, 

626 submodules=True, path=None): 

627 """ 

628 :return: 

629 ``True``, the repository is considered dirty. By default it will react 

630 like a git-status without untracked files, hence it is dirty if the 

631 index or the working copy have changes.""" 

632 if self._bare: 

633 # Bare repositories with no associated working directory are 

634 # always consired to be clean. 

635 return False 

636 

637 # start from the one which is fastest to evaluate 

638 default_args = ['--abbrev=40', '--full-index', '--raw'] 

639 if not submodules: 

640 default_args.append('--ignore-submodules') 

641 if path: 

642 default_args.append(path) 

643 if index: 

644 # diff index against HEAD 

645 if osp.isfile(self.index.path) and \

707 def blame_incremental(self, rev, file, **kwargs): 

708 """Iterator for blame information for the given file at the given revision. 

709 

710 Unlike .blame(), this does not return the actual file's contents, only 

711 a stream of BlameEntry tuples. 

712 

713 :param rev: revision specifier, see git-rev-parse for viable options. 

714 :return: lazy iterator of BlameEntry tuples, where the commit 

715 indicates the commit to blame for the line, and range 

716 indicates a span of line numbers in the resulting file. 

717 

718 If you combine all line number ranges outputted by this command, you 

719 should get a continuous range spanning all line numbers in the file. 

720 """ 

721 data = self.git.blame(rev, '--', file, p=True, incremental=True, stdout_as_string=False, **kwargs) 

722 commits = {} 

723 

724 stream = (line for line in data.split(b'\n') if line) 

725 while True: 

726 try: 

727 line = next(stream) # when exhausted, causes a StopIteration, terminating this function 

728 except StopIteration: 

729 return 

730 hexsha, orig_lineno, lineno, num_lines = line.split() 

731 lineno = int(lineno) 

732 num_lines = int(num_lines) 

733 orig_lineno = int(orig_lineno) 

734 if hexsha not in commits: 

735 # Now read the next few lines and build up a dict of properties 

736 # for this commit 

737 props = {} 

738 while True: 

739 try: 

740 line = next(stream) 

741 except StopIteration: 

742 return 

743 if line == b'boundary': 

744 # "boundary" indicates a root commit and occurs 

745 # instead of the "previous" tag 

746 continue 

747 

748 tag, value = line.split(b' ', 1) 

749 props[tag] = value 

750 if tag == b'filename': 

751 # "filename" formally terminates the entry for --incremental 

752 orig_filename = value 

753 break 

754 

755 c = Commit(self, hex_to_bin(hexsha), 

756 author=Actor(safe_decode(props[b'author']), 

757 safe_decode(props[b'author-mail'].lstrip(b'<').rstrip(b'>'))), 

758 authored_date=int(props[b'author-time']), 

759 committer=Actor(safe_decode(props[b'committer']), 

760 safe_decode(props[b'committer-mail'].lstrip(b'<').rstrip(b'>'))), 

761 committed_date=int(props[b'committer-time'])) 

762 commits[hexsha] = c 

763 else: 

764 # Discard all lines until we find "filename" which is 

765 # guaranteed to be the last line 

766 while True: 

767 try: 

768 line = next(stream) # will fail if we reach the EOF unexpectedly 

769 except StopIteration: 

770 return 

771 tag, value = line.split(b' ', 1) 

772 if tag == b'filename': 

773 orig_filename = value 

774 break 

775 

776 yield BlameEntry(commits[hexsha], 

777 range(lineno, lineno + num_lines), 

778 safe_decode(orig_filename), 

779 range(orig_lineno, orig_lineno + num_lines))

781 def blame(self, rev, file, incremental=False, **kwargs): 

782 """The blame information for the given file at the given revision. 

783 

784 :param rev: revision specifier, see git-rev-parse for viable options. 

785 :return: 

786 list: [git.Commit, list: [<line>]] 

787 A list of tuples associating a Commit object with a list of lines that 

788 changed within the given commit. The Commit objects will be given in order 

789 of appearance.""" 

790 if incremental: 

791 return self.blame_incremental(rev, file, **kwargs) 

792 

793 data = self.git.blame(rev, '--', file, p=True, stdout_as_string=False, **kwargs) 

794 commits = {} 

795 blames = [] 

796 info = None 

797 

798 keepends = True 

799 for line in data.splitlines(keepends): 

800 try: 

801 line = line.rstrip().decode(defenc) 

802 except UnicodeDecodeError: 

803 firstpart = '' 

804 is_binary = True 

805 else: 

806 # As we don't have an idea when the binary data ends, as it could contain multiple newlines 

807 # in the process. So we rely on being able to decode to tell us what is is. 

808 # This can absolutely fail even on text files, but even if it does, we should be fine treating it 

809 # as binary instead 

810 parts = self.re_whitespace.split(line, 1) 

811 firstpart = parts[0] 

812 is_binary = False 

813 # end handle decode of line 

814 

815 if self.re_hexsha_only.search(firstpart): 

816 # handles 

817 # 634396b2f541a9f2d58b00be1a07f0c358b999b3 1 1 7 - indicates blame-data start 

818 # 634396b2f541a9f2d58b00be1a07f0c358b999b3 2 2 - indicates 

819 # another line of blame with the same data 

820 digits = parts[-1].split(" ") 

821 if len(digits) == 3: 

822 info = {'id': firstpart} 

823 blames.append([None, []]) 

824 elif info['id'] != firstpart: 

825 info = {'id': firstpart} 

826 blames.append([commits.get(firstpart), []]) 

827 # END blame data initialization 

828 else: 

829 m = self.re_author_committer_start.search(firstpart) 

830 if m: 

831 # handles: 

832 # author Tom Preston-Werner 

833 # author-mail <tom@mojombo.com> 

834 # author-time 1192271832 

835 # author-tz -0700 

836 # committer Tom Preston-Werner 

837 # committer-mail <tom@mojombo.com> 

838 # committer-time 1192271832 

839 # committer-tz -0700 - IGNORED BY US 

840 role = m.group(0) 

841 if firstpart.endswith('-mail'): 

842 info["%s_email" % role] = parts[-1] 

843 elif firstpart.endswith('-time'): 

844 info["%s_date" % role] = int(parts[-1]) 

845 elif role == firstpart: 

846 info[role] = parts[-1] 

847 # END distinguish mail,time,name 

848 else: 

849 # handle 

850 # filename lib/grit.rb 

851 # summary add Blob 

852 # <and rest> 

853 if firstpart.startswith('filename'): 

854 info['filename'] = parts[-1] 

855 elif firstpart.startswith('summary'): 

856 info['summary'] = parts[-1] 

857 elif firstpart == '': 

858 if info: 

859 sha = info['id'] 

860 c = commits.get(sha) 

861 if c is None: 

862 c = Commit(self, hex_to_bin(sha), 

863 author=Actor._from_string(info['author'] + ' ' + info['author_email']), 

864 authored_date=info['author_date'], 

865 committer=Actor._from_string( 

866 info['committer'] + ' ' + info['committer_email']), 

867 committed_date=info['committer_date']) 

868 commits[sha] = c 

869 # END if commit objects needs initial creation 

870 if not is_binary: 

871 if line and line[0] == '\t': 

872 line = line[1:] 

873 else: 

874 # NOTE: We are actually parsing lines out of binary data, which can lead to the 

875 # binary being split up along the newline separator. We will append this to the blame 

876 # we are currently looking at, even though it should be concatenated with the last line 

877 # we have seen. 

878 pass 

879 # end handle line contents 

880 blames[-1][0] = c 

881 blames[-1][1].append(line) 

882 info = {'id': sha} 

883 # END if we collected commit info 

884 # END distinguish filename,summary,rest 

885 # END distinguish author|committer vs filename,summary,rest 

886 # END distinguish hexsha vs other information 

887 return blames
