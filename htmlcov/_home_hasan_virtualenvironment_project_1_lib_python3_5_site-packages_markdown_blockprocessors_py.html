21CORE MARKDOWN BLOCKPARSER 

22=========================================================================== 

23 

24This parser handles basic parsing of Markdown blocks. It doesn't concern 

25itself with inline elements such as **bold** or *italics*, but rather just 

26catches blocks, lists, quotes, etc. 

27 

28The BlockParser is made up of a bunch of BlockProcessors, each handling a 

29different type of block. Extensions may add/replace/remove BlockProcessors 

30as they need to alter how markdown blocks are parsed. 

31"""

42def build_block_parser(md, **kwargs): 

43 """ Build the default block parser used by Markdown. """ 

44 parser = BlockParser(md) 

45 parser.blockprocessors.register(EmptyBlockProcessor(parser), 'empty', 100) 

46 parser.blockprocessors.register(ListIndentProcessor(parser), 'indent', 90) 

47 parser.blockprocessors.register(CodeBlockProcessor(parser), 'code', 80) 

48 parser.blockprocessors.register(HashHeaderProcessor(parser), 'hashheader', 70) 

49 parser.blockprocessors.register(SetextHeaderProcessor(parser), 'setextheader', 60) 

50 parser.blockprocessors.register(HRProcessor(parser), 'hr', 50) 

51 parser.blockprocessors.register(OListProcessor(parser), 'olist', 40) 

52 parser.blockprocessors.register(UListProcessor(parser), 'ulist', 30) 

53 parser.blockprocessors.register(BlockQuoteProcessor(parser), 'quote', 20) 

54 parser.blockprocessors.register(ParagraphProcessor(parser), 'paragraph', 10) 

55 return parser

58class BlockProcessor: 

59 """ Base class for block processors. 

60 

61 Each subclass will provide the methods below to work with the source and 

62 tree. Each processor will need to define it's own ``test`` and ``run`` 

63 methods. The ``test`` method should return True or False, to indicate 

64 whether the current block should be processed by this processor. If the 

65 test passes, the parser will call the processors ``run`` method. 

66 

67 """ 

68 

69 def __init__(self, parser): 

70 self.parser = parser 

71 self.tab_length = parser.md.tab_length 

72 

73 def lastChild(self, parent): 

74 """ Return the last child of an etree element. """ 

75 if len(parent): 

76 return parent[-1] 

77 else: 

78 return None 

79 

80 def detab(self, text): 

81 """ Remove a tab from the front of each line of the given text. """ 

82 newtext = [] 

83 lines = text.split('\n') 

84 for line in lines: 

85 if line.startswith(' '*self.tab_length): 

86 newtext.append(line[self.tab_length:]) 

87 elif not line.strip(): 

88 newtext.append('') 

89 else: 

90 break 

91 return '\n'.join(newtext), '\n'.join(lines[len(newtext):]) 

92 

93 def looseDetab(self, text, level=1): 

94 """ Remove a tab from front of lines but allowing dedented lines. """ 

95 lines = text.split('\n') 

96 for i in range(len(lines)): 

97 if lines[i].startswith(' '*self.tab_length*level): 

98 lines[i] = lines[i][self.tab_length*level:] 

99 return '\n'.join(lines) 

100 

101 def test(self, parent, block): 

102 """ Test for block type. Must be overridden by subclasses. 

103 

104 As the parser loops through processors, it will call the ``test`` 

105 method on each to determine if the given block of text is of that 

106 type. This method must return a boolean ``True`` or ``False``. The 

107 actual method of testing is left to the needs of that particular 

108 block type. It could be as simple as ``block.startswith(some_string)`` 

109 or a complex regular expression. As the block type may be different 

110 depending on the parent of the block (i.e. inside a list), the parent 

111 etree element is also provided and may be used as part of the test. 

112 

113 Keywords: 

114 

115 * ``parent``: A etree element which will be the parent of the block. 

116 * ``block``: A block of text from the source which has been split at 

117 blank lines. 

118 """ 

119 pass # pragma: no cover 

120 

121 def run(self, parent, blocks): 

122 """ Run processor. Must be overridden by subclasses. 

123 

124 When the parser determines the appropriate type of a block, the parser 

125 will call the corresponding processor's ``run`` method. This method 

126 should parse the individual lines of the block and append them to 

127 the etree. 

128 

129 Note that both the ``parent`` and ``etree`` keywords are pointers 

130 to instances of the objects which should be edited in place. Each 

131 processor must make changes to the existing objects as there is no 

132 mechanism to return new/different objects to replace them. 

133 

134 This means that this method should be adding SubElements or adding text 

135 to the parent, and should remove (``pop``) or add (``insert``) items to 

136 the list of blocks. 

137 

138 Keywords: 

139 

140 * ``parent``: A etree element which is the parent of the current block. 

141 * ``blocks``: A list of all remaining blocks of the document. 

142 """ 

143 pass # pragma: no cover

146class ListIndentProcessor(BlockProcessor): 

147 """ Process children of list items. 

148 

149 Example: 

150 * a list item 

151 process this part 

152 

153 or this part 

154 

155 """ 

156 

157 ITEM_TYPES = ['li'] 

158 LIST_TYPES = ['ul', 'ol'] 

159 

160 def __init__(self, *args): 

161 super().__init__(*args) 

162 self.INDENT_RE = re.compile(r'^(([ ]{%s})+)' % self.tab_length) 

163 

164 def test(self, parent, block): 

165 return block.startswith(' '*self.tab_length) and \ 

166 not self.parser.state.isstate('detabbed') and \ 

167 (parent.tag in self.ITEM_TYPES or 

168 (len(parent) and parent[-1] is not None and 

169 (parent[-1].tag in self.LIST_TYPES))) 

170 

171 def run(self, parent, blocks): 

172 block = blocks.pop(0) 

173 level, sibling = self.get_level(parent, block) 

174 block = self.looseDetab(block, level) 

175 

176 self.parser.state.set('detabbed') 

177 if parent.tag in self.ITEM_TYPES: 

178 # It's possible that this parent has a 'ul' or 'ol' child list 

179 # with a member. If that is the case, then that should be the 

180 # parent. This is intended to catch the edge case of an indented 

181 # list whose first member was parsed previous to this point 

182 # see OListProcessor 

183 if len(parent) and parent[-1].tag in self.LIST_TYPES: 

184 self.parser.parseBlocks(parent[-1], [block]) 

185 else: 

186 # The parent is already a li. Just parse the child block. 

187 self.parser.parseBlocks(parent, [block]) 

188 elif sibling.tag in self.ITEM_TYPES: 

189 # The sibling is a li. Use it as parent. 

190 self.parser.parseBlocks(sibling, [block]) 

191 elif len(sibling) and sibling[-1].tag in self.ITEM_TYPES: 

192 # The parent is a list (``ol`` or ``ul``) which has children. 

193 # Assume the last child li is the parent of this block. 

194 if sibling[-1].text: 

195 # If the parent li has text, that text needs to be moved to a p 

196 # The p must be 'inserted' at beginning of list in the event 

197 # that other children already exist i.e.; a nested sublist. 

198 p = etree.Element('p') 

199 p.text = sibling[-1].text 

200 sibling[-1].text = '' 

201 sibling[-1].insert(0, p) 

202 self.parser.parseChunk(sibling[-1], block) 

203 else: 

204 self.create_item(sibling, block) 

205 self.parser.state.reset() 

206 

207 def create_item(self, parent, block): 

208 """ Create a new li and parse the block with it as the parent. """ 

209 li = etree.SubElement(parent, 'li') 

210 self.parser.parseBlocks(li, [block]) 

211 

212 def get_level(self, parent, block): 

213 """ Get level of indent based on list level. """ 

214 # Get indent level 

215 m = self.INDENT_RE.match(block) 

216 if m: 

217 indent_level = len(m.group(1))/self.tab_length 

218 else: 

219 indent_level = 0 

220 if self.parser.state.isstate('list'): 

221 # We're in a tightlist - so we already are at correct parent. 

222 level = 1 

223 else: 

224 # We're in a looselist - so we need to find parent. 

225 level = 0 

226 # Step through children of tree to find matching indent level. 

227 while indent_level > level: 

228 child = self.lastChild(parent) 

229 if (child is not None and 

230 (child.tag in self.LIST_TYPES or child.tag in self.ITEM_TYPES)): 

231 if child.tag in self.LIST_TYPES: 

232 level += 1 

233 parent = child 

234 else: 

235 # No more child levels. If we're short of indent_level, 

236 # we have a code block. So we stop here. 

237 break 

238 return level, parent

241class CodeBlockProcessor(BlockProcessor): 

242 """ Process code blocks. """ 

243 

244 def test(self, parent, block): 

245 return block.startswith(' '*self.tab_length) 

246 

247 def run(self, parent, blocks): 

248 sibling = self.lastChild(parent) 

249 block = blocks.pop(0) 

250 theRest = '' 

251 if (sibling is not None and sibling.tag == "pre" and 

252 len(sibling) and sibling[0].tag == "code"): 

253 # The previous block was a code block. As blank lines do not start 

254 # new code blocks, append this block to the previous, adding back 

255 # linebreaks removed from the split into a list. 

256 code = sibling[0] 

257 block, theRest = self.detab(block) 

258 code.text = util.AtomicString( 

259 '{}\n{}\n'.format(code.text, util.code_escape(block.rstrip())) 

260 ) 

261 else: 

262 # This is a new codeblock. Create the elements and insert text. 

263 pre = etree.SubElement(parent, 'pre') 

264 code = etree.SubElement(pre, 'code') 

265 block, theRest = self.detab(block) 

266 code.text = util.AtomicString('%s\n' % util.code_escape(block.rstrip())) 

267 if theRest: 

268 # This block contained unindented line(s) after the first indented 

269 # line. Insert these lines as the first block of the master blocks 

270 # list for future processing. 

271 blocks.insert(0, theRest)

274class BlockQuoteProcessor(BlockProcessor): 

275 

276 RE = re.compile(r'(^|\n)[ ]{0,3}>[ ]?(.*)') 

277 

278 def test(self, parent, block): 

279 return bool(self.RE.search(block)) 

280 

281 def run(self, parent, blocks): 

282 block = blocks.pop(0) 

283 m = self.RE.search(block) 

284 if m: 

285 before = block[:m.start()] # Lines before blockquote 

286 # Pass lines before blockquote in recursively for parsing forst. 

287 self.parser.parseBlocks(parent, [before]) 

288 # Remove ``> `` from beginning of each line. 

289 block = '\n'.join( 

290 [self.clean(line) for line in block[m.start():].split('\n')] 

291 ) 

292 sibling = self.lastChild(parent) 

293 if sibling is not None and sibling.tag == "blockquote": 

294 # Previous block was a blockquote so set that as this blocks parent 

295 quote = sibling 

296 else: 

297 # This is a new blockquote. Create a new parent element. 

298 quote = etree.SubElement(parent, 'blockquote') 

299 # Recursively parse block with blockquote as parent. 

300 # change parser state so blockquotes embedded in lists use p tags 

301 self.parser.state.set('blockquote') 

302 self.parser.parseChunk(quote, block) 

303 self.parser.state.reset() 

304 

305 def clean(self, line): 

306 """ Remove ``>`` from beginning of a line. """ 

307 m = self.RE.match(line) 

308 if line.strip() == ">": 

309 return "" 

310 elif m: 

311 return m.group(2) 

312 else: 

313 return line

316class OListProcessor(BlockProcessor): 

317 """ Process ordered list blocks. """ 

318 

319 TAG = 'ol' 

320 # The integer (python string) with which the lists starts (default=1) 

321 # Eg: If list is intialized as) 

322 # 3. Item 

323 # The ol tag will get starts="3" attribute 

324 STARTSWITH = '1' 

325 # Lazy ol - ignore startswith 

326 LAZY_OL = True 

327 # List of allowed sibling tags. 

328 SIBLING_TAGS = ['ol', 'ul'] 

329 

330 def __init__(self, parser): 

331 super().__init__(parser) 

332 # Detect an item (``1. item``). ``group(1)`` contains contents of item. 

333 self.RE = re.compile(r'^[ ]{0,%d}\d+\.[ ]+(.*)' % (self.tab_length - 1)) 

334 # Detect items on secondary lines. they can be of either list type. 

335 self.CHILD_RE = re.compile(r'^[ ]{0,%d}((\d+\.)|[*+-])[ ]+(.*)' % 

336 (self.tab_length - 1)) 

337 # Detect indented (nested) items of either type 

338 self.INDENT_RE = re.compile(r'^[ ]{%d,%d}((\d+\.)|[*+-])[ ]+.*' % 

339 (self.tab_length, self.tab_length * 2 - 1)) 

340 

341 def test(self, parent, block): 

342 return bool(self.RE.match(block)) 

343 

344 def run(self, parent, blocks): 

345 # Check fr multiple items in one block. 

346 items = self.get_items(blocks.pop(0)) 

347 sibling = self.lastChild(parent) 

348 

349 if sibling is not None and sibling.tag in self.SIBLING_TAGS: 

350 # Previous block was a list item, so set that as parent 

351 lst = sibling 

352 # make sure previous item is in a p- if the item has text, 

353 # then it isn't in a p 

354 if lst[-1].text: 

355 # since it's possible there are other children for this 

356 # sibling, we can't just SubElement the p, we need to 

357 # insert it as the first item. 

358 p = etree.Element('p') 

359 p.text = lst[-1].text 

360 lst[-1].text = '' 

361 lst[-1].insert(0, p) 

362 # if the last item has a tail, then the tail needs to be put in a p 

363 # likely only when a header is not followed by a blank line 

364 lch = self.lastChild(lst[-1]) 

365 if lch is not None and lch.tail: 

366 p = etree.SubElement(lst[-1], 'p') 

367 p.text = lch.tail.lstrip() 

368 lch.tail = '' 

369 

370 # parse first block differently as it gets wrapped in a p. 

371 li = etree.SubElement(lst, 'li') 

372 self.parser.state.set('looselist') 

373 firstitem = items.pop(0) 

374 self.parser.parseBlocks(li, [firstitem]) 

375 self.parser.state.reset() 

376 elif parent.tag in ['ol', 'ul']: 

377 # this catches the edge case of a multi-item indented list whose 

378 # first item is in a blank parent-list item: 

379 # * * subitem1 

380 # * subitem2 

381 # see also ListIndentProcessor 

382 lst = parent 

383 else: 

384 # This is a new list so create parent with appropriate tag. 

385 lst = etree.SubElement(parent, self.TAG) 

386 # Check if a custom start integer is set 

387 if not self.LAZY_OL and self.STARTSWITH != '1': 

388 lst.attrib['start'] = self.STARTSWITH 

389 

390 self.parser.state.set('list') 

391 # Loop through items in block, recursively parsing each with the 

392 # appropriate parent. 

393 for item in items: 

394 if item.startswith(' '*self.tab_length): 

395 # Item is indented. Parse with last item as parent 

396 self.parser.parseBlocks(lst[-1], [item]) 

397 else: 

398 # New item. Create li and parse with it as parent 

399 li = etree.SubElement(lst, 'li') 

400 self.parser.parseBlocks(li, [item]) 

401 self.parser.state.reset() 

402 

403 def get_items(self, block): 

404 """ Break a block into list items. """ 

405 items = [] 

406 for line in block.split('\n'): 

407 m = self.CHILD_RE.match(line) 

408 if m: 

409 # This is a new list item 

410 # Check first item for the start index 

411 if not items and self.TAG == 'ol': 

412 # Detect the integer value of first list item 

413 INTEGER_RE = re.compile(r'(\d+)') 

414 self.STARTSWITH = INTEGER_RE.match(m.group(1)).group() 

415 # Append to the list 

416 items.append(m.group(3)) 

417 elif self.INDENT_RE.match(line): 

418 # This is an indented (possibly nested) item. 

419 if items[-1].startswith(' '*self.tab_length): 

420 # Previous item was indented. Append to that item. 

421 items[-1] = '{}\n{}'.format(items[-1], line) 

422 else: 

423 items.append(line) 

424 else: 

425 # This is another line of previous item. Append to that item. 

426 items[-1] = '{}\n{}'.format(items[-1], line) 

427 return items

441class HashHeaderProcessor(BlockProcessor): 

442 """ Process Hash Headers. """ 

443 

444 # Detect a header at start of any line in block 

445 RE = re.compile(r'(?:^|\n)(?P<level>#{1,6})(?P<header>(?:\\.|[^\\])*?)#*(?:\n|$)') 

446 

447 def test(self, parent, block): 

448 return bool(self.RE.search(block)) 

449 

450 def run(self, parent, blocks): 

451 block = blocks.pop(0) 

452 m = self.RE.search(block) 

453 if m: 

454 before = block[:m.start()] # All lines before header 

455 after = block[m.end():] # All lines after header 

456 if before: 

457 # As the header was not the first line of the block and the 

458 # lines before the header must be parsed first, 

459 # recursively parse this lines as a block. 

460 self.parser.parseBlocks(parent, [before]) 

461 # Create header using named groups from RE 

462 h = etree.SubElement(parent, 'h%d' % len(m.group('level'))) 

463 h.text = m.group('header').strip() 

464 if after: 

465 # Insert remaining lines as first block for future parsing. 

466 blocks.insert(0, after) 

467 else: # pragma: no cover 

468 # This should never happen, but just in case... 

469 logger.warn("We've got a problem header: %r" % block)

472class SetextHeaderProcessor(BlockProcessor): 

473 """ Process Setext-style Headers. """ 

474 

475 # Detect Setext-style header. Must be first 2 lines of block. 

476 RE = re.compile(r'^.*?\n[=-]+[ ]*(\n|$)', re.MULTILINE) 

477 

478 def test(self, parent, block): 

479 return bool(self.RE.match(block)) 

480 

481 def run(self, parent, blocks): 

482 lines = blocks.pop(0).split('\n') 

483 # Determine level. ``=`` is 1 and ``-`` is 2. 

484 if lines[1].startswith('='): 

485 level = 1 

486 else: 

487 level = 2 

488 h = etree.SubElement(parent, 'h%d' % level) 

489 h.text = lines[0].strip() 

490 if len(lines) > 2: 

491 # Block contains additional lines. Add to master blocks for later. 

492 blocks.insert(0, '\n'.join(lines[2:]))

495class HRProcessor(BlockProcessor): 

496 """ Process Horizontal Rules. """ 

497 

498 RE = r'^[ ]{0,3}((-+[ ]{0,2}){3,}|(_+[ ]{0,2}){3,}|(\*+[ ]{0,2}){3,})[ ]*' 

499 # Detect hr on any line of a block. 

500 SEARCH_RE = re.compile(RE, re.MULTILINE) 

501 

502 def test(self, parent, block): 

503 m = self.SEARCH_RE.search(block) 

504 # No atomic grouping in python so we simulate it here for performance. 

505 # The regex only matches what would be in the atomic group - the HR. 

506 # Then check if we are at end of block or if next char is a newline. 

507 if m and (m.end() == len(block) or block[m.end()] == '\n'): 

508 # Save match object on class instance so we can use it later. 

509 self.match = m 

510 return True 

511 return False 

512 

513 def run(self, parent, blocks): 

514 block = blocks.pop(0) 

515 match = self.match 

516 # Check for lines in block before hr. 

517 prelines = block[:match.start()].rstrip('\n') 

518 if prelines: 

519 # Recursively parse lines before hr so they get parsed first. 

520 self.parser.parseBlocks(parent, [prelines]) 

521 # create hr 

522 etree.SubElement(parent, 'hr') 

523 # check for lines in block after hr. 

524 postlines = block[match.end():].lstrip('\n') 

525 if postlines: 

526 # Add lines after hr to master blocks for later parsing. 

527 blocks.insert(0, postlines)

530class EmptyBlockProcessor(BlockProcessor): 

531 """ Process blocks that are empty or start with an empty line. """ 

532 

533 def test(self, parent, block): 

534 return not block or block.startswith('\n') 

535 

536 def run(self, parent, blocks): 

537 block = blocks.pop(0) 

538 filler = '\n\n' 

539 if block: 

540 # Starts with empty line 

541 # Only replace a single line. 

542 filler = '\n' 

543 # Save the rest for later. 

544 theRest = block[1:] 

545 if theRest: 

546 # Add remaining lines to master blocks for later. 

547 blocks.insert(0, theRest) 

548 sibling = self.lastChild(parent) 

549 if (sibling is not None and sibling.tag == 'pre' and 

550 len(sibling) and sibling[0].tag == 'code'): 

551 # Last block is a codeblock. Append to preserve whitespace. 

552 sibling[0].text = util.AtomicString( 

553 '{}{}'.format(sibling[0].text, filler)

557class ParagraphProcessor(BlockProcessor): 

558 """ Process Paragraph blocks. """ 

559 

560 def test(self, parent, block): 

561 return True 

562 

563 def run(self, parent, blocks): 

564 block = blocks.pop(0) 

565 if block.strip(): 

566 # Not a blank block. Add to parent, otherwise throw it away. 

567 if self.parser.state.isstate('list'): 

568 # The parent is a tight-list. 

569 # 

570 # Check for any children. This will likely only happen in a 

571 # tight-list when a header isn't followed by a blank line. 

572 # For example: 

573 # 

574 # * # Header 

575 # Line 2 of list item - not part of header. 

576 sibling = self.lastChild(parent) 

577 if sibling is not None: 

578 # Insetrt after sibling. 

579 if sibling.tail: 

580 sibling.tail = '{}\n{}'.format(sibling.tail, block) 

581 else: 

582 sibling.tail = '\n%s' % block 

583 else: 

584 # Append to parent.text 

585 if parent.text: 

586 parent.text = '{}\n{}'.format(parent.text, block) 

587 else: 

588 parent.text = block.lstrip() 

589 else: 

590 # Create a regular paragraph 

591 p = etree.SubElement(parent, 'p') 

592 p.text = block.lstrip()
