148def _databases_recursive(database, output): 

149 """Fill output list with database from db, in order. Deals with Loose, Packed 

150 and compound databases.""" 

151 if isinstance(database, CompoundDB): 

152 dbs = database.databases() 

153 output.extend(db for db in dbs if not isinstance(db, CompoundDB)) 

154 for cdb in (db for db in dbs if isinstance(db, CompoundDB)): 

155 _databases_recursive(cdb, output) 

156 else: 

157 output.append(database)

161class CompoundDB(ObjectDBR, LazyMixin, CachingDB): 

162 

163 """A database which delegates calls to sub-databases. 

164 

165 Databases are stored in the lazy-loaded _dbs attribute. 

166 Define _set_cache_ to update it with your databases""" 

167 

168 def _set_cache_(self, attr): 

169 if attr == '_dbs': 

170 self._dbs = list() 

171 elif attr == '_db_cache': 

172 self._db_cache = dict() 

173 else: 

174 super(CompoundDB, self)._set_cache_(attr) 

175 

176 def _db_query(self, sha): 

177 """:return: database containing the given 20 byte sha 

178 :raise BadObject:""" 

179 # most databases use binary representations, prevent converting 

180 # it every time a database is being queried 

181 try: 

182 return self._db_cache[sha] 

183 except KeyError: 

184 pass 

185 # END first level cache 

186 

187 for db in self._dbs: 

188 if db.has_object(sha): 

189 self._db_cache[sha] = db 

190 return db 

191 # END for each database 

192 raise BadObject(sha) 

193 

194 #{ ObjectDBR interface 

195 

196 def has_object(self, sha): 

197 try: 

198 self._db_query(sha) 

199 return True 

200 except BadObject: 

201 return False 

202 # END handle exceptions 

203 

204 def info(self, sha): 

205 return self._db_query(sha).info(sha) 

206 

207 def stream(self, sha): 

208 return self._db_query(sha).stream(sha) 

209 

210 def size(self): 

211 """:return: total size of all contained databases""" 

212 return reduce(lambda x, y: x + y, (db.size() for db in self._dbs), 0)

225 def update_cache(self, force=False): 

226 # something might have changed, clear everything 

227 self._db_cache.clear() 

228 stat = False 

229 for db in self._dbs: 

230 if isinstance(db, CachingDB): 

231 stat |= db.update_cache(force) 

232 # END if is caching db 

233 # END for each database to update 

234 return stat

236 def partial_to_complete_sha_hex(self, partial_hexsha): 

237 """ 

238 :return: 20 byte binary sha1 from the given less-than-40 byte hexsha (bytes or str) 

239 :param partial_hexsha: hexsha with less than 40 byte 

240 :raise AmbiguousObjectName: """ 

241 databases = list() 

242 _databases_recursive(self, databases) 

243 partial_hexsha = force_text(partial_hexsha) 

244 len_partial_hexsha = len(partial_hexsha) 

245 if len_partial_hexsha % 2 != 0: 

246 partial_binsha = hex_to_bin(partial_hexsha + "0") 

247 else: 

248 partial_binsha = hex_to_bin(partial_hexsha) 

249 # END assure successful binary conversion 

250 

251 candidate = None 

252 for db in databases: 

253 full_bin_sha = None 

254 try: 

255 if hasattr(db, 'partial_to_complete_sha_hex'): 

256 full_bin_sha = db.partial_to_complete_sha_hex(partial_hexsha) 

257 else: 

258 full_bin_sha = db.partial_to_complete_sha(partial_binsha, len_partial_hexsha) 

259 # END handle database type 

260 except BadObject: 

261 continue 

262 # END ignore bad objects 

263 if full_bin_sha: 

264 if candidate and candidate != full_bin_sha: 

265 raise AmbiguousObjectName(partial_hexsha) 

266 candidate = full_bin_sha 

267 # END handle candidate 

268 # END for each db 

269 if not candidate: 

270 raise BadObject(partial_binsha) 

271 return candidate
