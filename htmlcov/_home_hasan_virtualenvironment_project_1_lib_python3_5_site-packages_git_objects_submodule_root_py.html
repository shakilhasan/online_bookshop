64 def update(self, previous_commit=None, recursive=True, force_remove=False, init=True, 

65 to_latest_revision=False, progress=None, dry_run=False, force_reset=False, 

66 keep_going=False): 

67 """Update the submodules of this repository to the current HEAD commit. 

68 This method behaves smartly by determining changes of the path of a submodules 

69 repository, next to changes to the to-be-checked-out commit or the branch to be 

70 checked out. This works if the submodules ID does not change. 

71 Additionally it will detect addition and removal of submodules, which will be handled 

72 gracefully. 

73 

74 :param previous_commit: If set to a commit'ish, the commit we should use 

75 as the previous commit the HEAD pointed to before it was set to the commit it points to now. 

76 If None, it defaults to HEAD@{1} otherwise 

77 :param recursive: if True, the children of submodules will be updated as well 

78 using the same technique 

79 :param force_remove: If submodules have been deleted, they will be forcibly removed. 

80 Otherwise the update may fail if a submodule's repository cannot be deleted as 

81 changes have been made to it (see Submodule.update() for more information) 

82 :param init: If we encounter a new module which would need to be initialized, then do it. 

83 :param to_latest_revision: If True, instead of checking out the revision pointed to 

84 by this submodule's sha, the checked out tracking branch will be merged with the 

85 latest remote branch fetched from the repository's origin. 

86 Unless force_reset is specified, a local tracking branch will never be reset into its past, therefore 

87 the remote branch must be in the future for this to have an effect. 

88 :param force_reset: if True, submodules may checkout or reset their branch even if the repository has 

89 pending changes that would be overwritten, or if the local tracking branch is in the future of the 

90 remote tracking branch and would be reset into its past. 

91 :param progress: RootUpdateProgress instance or None if no progress should be sent 

92 :param dry_run: if True, operations will not actually be performed. Progress messages 

93 will change accordingly to indicate the WOULD DO state of the operation. 

94 :param keep_going: if True, we will ignore but log all errors, and keep going recursively. 

95 Unless dry_run is set as well, keep_going could cause subsequent/inherited errors you wouldn't see 

96 otherwise. 

97 In conjunction with dry_run, it can be useful to anticipate all errors when updating submodules 

98 :return: self"""

121 raise IndexError 

122 # END handle initial commit 

123 except IndexError:

125 previous_commit = cur_commit

143 if i == 0:

145 # END handle begin

148 # of previous module. Trigger the cache to be updated before that 

149 progress.update(op, i, len_rrsm, prefix + "Removing submodule %r at %s" % (rsm.name, rsm.abspath)) 

150 rsm._parent_commit = repo.head.commit 

151 rsm.remove(configuration=False, module=True, force=force_remove, dry_run=dry_run)

153 if i == len_rrsm - 1: 

154 op |= END 

155 # END handle end 

156 progress.update(op, i, len_rrsm, prefix + "Done removing submodule %r" % rsm.name) 

157 # END for each removed submodule

180 # END handle path changes

232 raise InvalidGitRepositoryError(

234 # END handle one single remote 

235 # END handle check we found a remote

237 orig_name = rmt_for_deletion.name 

238 smm.delete_remote(rmt_for_deletion) 

239 # NOTE: Currently we leave tags from the deleted remotes 

240 # as well as separate tracking branches in the possibly totally 

241 # changed repository ( someone could have changed the url to 

242 # another project ). At some point, one might want to clean 

243 # it up, but the danger is high to remove stuff the user 

244 # has added explicitly 

245 

246 # rename the new remote back to what it was 

247 smr.rename(orig_name) 

248 

249 # early on, we verified that the our current tracking branch

252 # tracking branch. 

253 smsha = sm.binsha 

254 found = False 

255 rref = smr.refs[self.branch_name] 

256 for c in rref.commit.traverse(): 

257 if c.binsha == smsha: 

258 found = True 

259 break 

260 # END traverse all commits in search for sha 

261 # END for each commit 

262 

263 if not found: 

264 # adjust our internal binsha to use the one of the remote 

265 # this way, it will be checked out in the next step 

266 # This will change the submodule relative to us, so 

267 # the user will be able to commit the change easily 

268 log.warning("Current sha %s was not contained in the tracking\ 

269 branch at the new remote, setting it the the remote's tracking branch", sm.hexsha) 

270 sm.binsha = rref.commit.binsha 

271 # END reset binsha 

272 

273 # NOTE: All checkout is performed by the base implementation of update 

274 # END handle dry_run 

275 progress.update( 

276 END | URLCHANGE, i, len_csms, prefix + "Done adjusting url of submodule %r" % (sm.name)) 

277 # END skip remote handling if new url already exists in module 

278 # END handle url 

279 

280 # HANDLE PATH CHANGES 

281 ##################### 

282 if sm.branch_path != psm.branch_path: 

283 # finally, create a new tracking branch which tracks the 

284 # new remote branch 

285 progress.update(BEGIN | BRANCHCHANGE, i, len_csms, prefix + 

286 "Changing branch of submodule %r from %s to %s" 

287 % (sm.name, psm.branch_path, sm.branch_path)) 

288 if not dry_run: 

289 smm = sm.module() 

290 smmr = smm.remotes 

291 # As the branch might not exist yet, we will have to fetch all remotes to be sure ... . 

292 for remote in smmr: 

293 remote.fetch(progress=progress) 

294 # end for each remote 

295

297 tbr = git.Head.create(smm, sm.branch_name, logmsg='branch: Created from HEAD')

300 tbr = git.Head(smm, sm.branch_path)

306 # checkout out commit, we don't do any harm.

316 except Exception as err:

318 raise 

319 log.error(str(err))

323 ###################################### 

324 for sm in sms: 

325 # update the submodule using the default method 

326 sm.update(recursive=False, init=init, to_latest_revision=to_latest_revision, 

327 progress=progress, dry_run=dry_run, force=force_reset, keep_going=keep_going) 

328
