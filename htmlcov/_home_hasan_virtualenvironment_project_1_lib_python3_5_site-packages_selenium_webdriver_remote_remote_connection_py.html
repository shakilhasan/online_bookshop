106 def __init__(self, remote_server_addr, keep_alive=False, resolve_ip=True): 

107 # Attempt to resolve the hostname and get an IP address. 

108 self.keep_alive = keep_alive 

109 parsed_url = parse.urlparse(remote_server_addr) 

110 if parsed_url.hostname and resolve_ip: 

111 port = parsed_url.port or None 

112 if parsed_url.scheme == "https": 

113 ip = parsed_url.hostname 

114 elif port and not common_utils.is_connectable(port, parsed_url.hostname): 

115 ip = None 

116 LOGGER.info('Could not connect to port {} on host ' 

117 '{}'.format(port, parsed_url.hostname)) 

118 else: 

119 ip = common_utils.find_connectable_ip(parsed_url.hostname, 

120 port=port) 

121 if ip: 

122 netloc = ip 

123 if parsed_url.port: 

124 netloc = common_utils.join_host_port(netloc, 

125 parsed_url.port) 

126 if parsed_url.username: 

127 auth = parsed_url.username 

128 if parsed_url.password: 

129 auth += ':%s' % parsed_url.password 

130 netloc = '%s@%s' % (auth, netloc) 

131 remote_server_addr = parse.urlunparse( 

132 (parsed_url.scheme, netloc, parsed_url.path, 

133 parsed_url.params, parsed_url.query, parsed_url.fragment)) 

134 else: 

135 LOGGER.info('Could not get IP address for host: %s' % 

136 parsed_url.hostname) 

137 

138 self._url = remote_server_addr 

139 if keep_alive: 

140 self._conn = urllib3.PoolManager(timeout=self._timeout)

355 def execute(self, command, params): 

356 """ 

357 Send a command to the remote server. 

358 

359 Any path subtitutions required for the URL mapped to the command should be 

360 included in the command parameters. 

361 

362 :Args: 

363 - command - A string specifying the command to execute. 

364 - params - A dictionary of named parameters to send with the command as 

365 its JSON payload. 

366 """ 

367 command_info = self._commands[command] 

368 assert command_info is not None, 'Unrecognised command %s' % command 

369 path = string.Template(command_info[1]).substitute(params) 

370 if hasattr(self, 'w3c') and self.w3c and isinstance(params, dict) and 'sessionId' in params: 

371 del params['sessionId'] 

372 data = utils.dump_json(params) 

373 url = '%s%s' % (self._url, path) 

374 return self._request(command_info[0], url, body=data)

376 def _request(self, method, url, body=None): 

377 """ 

378 Send an HTTP request to the remote server. 

379 

380 :Args: 

381 - method - A string for the HTTP method to send the request with. 

382 - url - A string for the URL to send the request to. 

383 - body - A string for request body. Ignored unless method is POST or PUT. 

384 

385 :Returns: 

386 A dictionary with the server's parsed JSON response. 

387 """ 

388 LOGGER.debug('%s %s %s' % (method, url, body)) 

389 

390 parsed_url = parse.urlparse(url) 

391 headers = self.get_remote_connection_headers(parsed_url, self.keep_alive) 

392 resp = None 

393 if body and method != 'POST' and method != 'PUT': 

394 body = None 

395 

396 if self.keep_alive: 

397 resp = self._conn.request(method, url, body=body, headers=headers) 

398 

399 statuscode = resp.status 

400 else: 

401 http = urllib3.PoolManager(timeout=self._timeout) 

402 resp = http.request(method, url, body=body, headers=headers) 

403 

404 statuscode = resp.status 

405 if not hasattr(resp, 'getheader'): 

406 if hasattr(resp.headers, 'getheader'): 

407 resp.getheader = lambda x: resp.headers.getheader(x) 

408 elif hasattr(resp.headers, 'get'): 

409 resp.getheader = lambda x: resp.headers.get(x) 

410 

411 data = resp.data.decode('UTF-8') 

412 try: 

413 if 300 <= statuscode < 304: 

414 return self._request('GET', resp.getheader('location')) 

415 if 399 < statuscode <= 500: 

416 return {'status': statuscode, 'value': data} 

417 content_type = [] 

418 if resp.getheader('Content-Type') is not None: 

419 content_type = resp.getheader('Content-Type').split(';') 

420 if not any([x.startswith('image/png') for x in content_type]): 

421 

422 try: 

423 data = utils.load_json(data.strip()) 

424 except ValueError: 

425 if 199 < statuscode < 300: 

426 status = ErrorCode.SUCCESS 

427 else: 

428 status = ErrorCode.UNKNOWN_ERROR 

429 return {'status': status, 'value': data.strip()} 

430 

431 # Some of the drivers incorrectly return a response 

432 # with no 'value' field when they should return null. 

433 if 'value' not in data: 

434 data['value'] = None 

435 return data 

436 else: 

437 data = {'status': 0, 'value': data} 

438 return data 

439 finally: 

440 LOGGER.debug("Finished Request") 

441 resp.close()
