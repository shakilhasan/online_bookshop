29def mode_str_to_int(modestr): 

30 """ 

31 :param modestr: string like 755 or 644 or 100644 - only the last 6 chars will be used 

32 :return: 

33 String identifying a mode compatible to the mode methods ids of the 

34 stat module regarding the rwx permissions for user, group and other, 

35 special flags and file system flags, i.e. whether it is a symlink 

36 for example.""" 

37 mode = 0 

38 for iteration, char in enumerate(reversed(modestr[-6:])): 

39 mode += int(char) << iteration * 3 

40 # END for each char

68def utctz_to_altz(utctz): 

69 """we convert utctz to the timezone in seconds, it is the format time.altzone 

70 returns. Git stores it as UTC timezone which has the opposite sign as well, 

71 which explains the -1 * ( that was made explicit here ) 

72 :param utctz: git utc timezone string, i.e. +0200""" 

73 return -1 * int(float(utctz) / 100 * 3600)

76def altz_to_utctz_str(altz): 

77 """As above, but inverses the operation, returning a string that can be used 

78 in commit objects""" 

79 utci = -1 * int((float(altz) / 3600) * 100) 

80 utcs = str(abs(utci)) 

81 utcs = "0" * (4 - len(utcs)) + utcs 

82 prefix = (utci < 0 and '-') or '+' 

83 return prefix + utcs

134def parse_date(string_date): 

135 """ 

136 Parse the given date as one of the following 

137 

138 * aware datetime instance 

139 * Git internal format: timestamp offset 

140 * RFC 2822: Thu, 07 Apr 2005 22:13:13 +0200. 

141 * ISO 8601 2005-04-07T22:13:13 

142 The T can be a space as well 

143 

144 :return: Tuple(int(timestamp_UTC), int(offset)), both in seconds since epoch 

145 :raise ValueError: If the format could not be understood 

146 :note: Date can also be YYYY.MM.DD, MM/DD/YYYY and DD.MM.YYYY. 

147 """ 

148 if isinstance(string_date, datetime) and string_date.tzinfo: 

149 offset = -int(string_date.utcoffset().total_seconds()) 

150 return int(string_date.astimezone(utc).timestamp()), offset 

151 

152 # git time 

153 try: 

154 if string_date.count(' ') == 1 and string_date.rfind(':') == -1: 

155 timestamp, offset = string_date.split() 

156 if timestamp.startswith('@'): 

157 timestamp = timestamp[1:] 

158 timestamp = int(timestamp) 

159 return timestamp, utctz_to_altz(verify_utctz(offset)) 

160 else: 

161 offset = "+0000" # local time by default 

162 if string_date[-5] in '-+': 

163 offset = verify_utctz(string_date[-5:]) 

164 string_date = string_date[:-6] # skip space as well 

165 # END split timezone info 

166 offset = utctz_to_altz(offset) 

167 

168 # now figure out the date and time portion - split time 

169 date_formats = [] 

170 splitter = -1 

171 if ',' in string_date: 

172 date_formats.append("%a, %d %b %Y") 

173 splitter = string_date.rfind(' ') 

174 else: 

175 # iso plus additional 

176 date_formats.append("%Y-%m-%d") 

177 date_formats.append("%Y.%m.%d") 

178 date_formats.append("%m/%d/%Y") 

179 date_formats.append("%d.%m.%Y") 

180 

181 splitter = string_date.rfind('T') 

182 if splitter == -1: 

183 splitter = string_date.rfind(' ') 

184 # END handle 'T' and ' ' 

185 # END handle rfc or iso 

186 

187 assert splitter > -1 

188 

189 # split date and time 

190 time_part = string_date[splitter + 1:] # skip space 

191 date_part = string_date[:splitter] 

192 

193 # parse time 

194 tstruct = time.strptime(time_part, "%H:%M:%S") 

195 

196 for fmt in date_formats: 

197 try: 

198 dtstruct = time.strptime(date_part, fmt) 

199 utctime = calendar.timegm((dtstruct.tm_year, dtstruct.tm_mon, dtstruct.tm_mday, 

200 tstruct.tm_hour, tstruct.tm_min, tstruct.tm_sec, 

201 dtstruct.tm_wday, dtstruct.tm_yday, tstruct.tm_isdst)) 

202 return int(utctime), offset 

203 except ValueError: 

204 continue 

205 # END exception handling 

206 # END for each fmt 

207 

208 # still here ? fail 

209 raise ValueError("no format matched") 

210 # END handle format 

211 except Exception as e: 

212 raise ValueError("Unsupported date format: %s" % string_date) from e 

213 # END handle exceptions

221def parse_actor_and_date(line): 

222 """Parse out the actor (author or committer) info from a line like:: 

223 

224 author Tom Preston-Werner <tom@mojombo.com> 1191999972 -0700 

225 

226 :return: [Actor, int_seconds_since_epoch, int_timezone_offset]""" 

227 actor, epoch, offset = '', 0, 0 

228 m = _re_actor_epoch.search(line) 

229 if m: 

230 actor, epoch, offset = m.groups() 

231 else: 

232 m = _re_only_actor.search(line) 

233 actor = m.group(1) if m else line or '' 

234 return (Actor._from_string(actor), int(epoch), utctz_to_altz(offset))

283 def traverse(self, predicate=lambda i, d: True, 

284 prune=lambda i, d: False, depth=-1, branch_first=True, 

285 visit_once=True, ignore_self=1, as_edge=False): 

286 """:return: iterator yielding of items found when traversing self 

287 

288 :param predicate: f(i,d) returns False if item i at depth d should not be included in the result 

289 

290 :param prune: 

291 f(i,d) return True if the search should stop at item i at depth d. 

292 Item i will not be returned. 

293 

294 :param depth: 

295 define at which level the iteration should not go deeper 

296 if -1, there is no limit 

297 if 0, you would effectively only get self, the root of the iteration 

298 i.e. if 1, you would only get the first level of predecessors/successors 

299 

300 :param branch_first: 

301 if True, items will be returned branch first, otherwise depth first 

302 

303 :param visit_once: 

304 if True, items will only be returned once, although they might be encountered 

305 several times. Loops are prevented that way. 

306 

307 :param ignore_self: 

308 if True, self will be ignored and automatically pruned from 

309 the result. Otherwise it will be the first item to be returned. 

310 If as_edge is True, the source of the first edge is None 

311 

312 :param as_edge: 

313 if True, return a pair of items, first being the source, second the 

314 destination, i.e. tuple(src, dest) with the edge spanning from 

315 source to destination""" 

316 visited = set() 

317 stack = Deque() 

318 stack.append((0, self, None)) # self is always depth level 0 

319 

320 def addToStack(stack, item, branch_first, depth): 

321 lst = self._get_intermediate_items(item) 

322 if not lst: 

323 return 

324 if branch_first: 

325 stack.extendleft((depth, i, item) for i in lst) 

326 else: 

327 reviter = ((depth, lst[i], item) for i in range(len(lst) - 1, -1, -1)) 

328 stack.extend(reviter) 

329 # END addToStack local method 

330 

331 while stack: 

332 d, item, src = stack.pop() # depth of item, item, item_source 

333 

334 if visit_once and item in visited: 

335 continue 

336 

337 if visit_once: 

338 visited.add(item) 

339 

340 rval = (as_edge and (src, item)) or item 

341 if prune(rval, d): 

342 continue 

343 

344 skipStartItem = ignore_self and (item is self) 

345 if not skipStartItem and predicate(rval, d): 

346 yield rval 

347 

348 # only continue to next level if this is appropriate ! 

349 nd = d + 1 

350 if depth > -1 and nd > depth: 

351 continue 

352 

353 addToStack(stack, item, branch_first, nd) 

354 # END for each item on work stack
