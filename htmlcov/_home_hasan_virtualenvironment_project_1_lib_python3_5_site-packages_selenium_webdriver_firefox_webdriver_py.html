50 def __init__(self, firefox_profile=None, firefox_binary=None, 

51 timeout=30, capabilities=None, proxy=None, 

52 executable_path="geckodriver", options=None, 

53 service_log_path="geckodriver.log", firefox_options=None, 

54 service_args=None, desired_capabilities=None, log_path=None, 

55 keep_alive=True): 

56 """Starts a new local session of Firefox. 

57 

58 Based on the combination and specificity of the various keyword 

59 arguments, a capabilities dictionary will be constructed that 

60 is passed to the remote end. 

61 

62 The keyword arguments given to this constructor are helpers to 

63 more easily allow Firefox WebDriver sessions to be customised 

64 with different options. They are mapped on to a capabilities 

65 dictionary that is passed on to the remote end. 

66 

67 As some of the options, such as `firefox_profile` and 

68 `options.profile` are mutually exclusive, precedence is 

69 given from how specific the setting is. `capabilities` is the 

70 least specific keyword argument, followed by `options`, 

71 followed by `firefox_binary` and `firefox_profile`. 

72 

73 In practice this means that if `firefox_profile` and 

74 `options.profile` are both set, the selected profile 

75 instance will always come from the most specific variable. 

76 In this case that would be `firefox_profile`. This will result in 

77 `options.profile` to be ignored because it is considered 

78 a less specific setting than the top-level `firefox_profile` 

79 keyword argument. Similarily, if you had specified a 

80 `capabilities["moz:firefoxOptions"]["profile"]` Base64 string, 

81 this would rank below `options.profile`. 

82 

83 :param firefox_profile: Instance of ``FirefoxProfile`` object 

84 or a string. If undefined, a fresh profile will be created 

85 in a temporary location on the system. 

86 :param firefox_binary: Instance of ``FirefoxBinary`` or full 

87 path to the Firefox binary. If undefined, the system default 

88 Firefox installation will be used. 

89 :param timeout: Time to wait for Firefox to launch when using 

90 the extension connection. 

91 :param capabilities: Dictionary of desired capabilities. 

92 :param proxy: The proxy settings to us when communicating with 

93 Firefox via the extension connection. 

94 :param executable_path: Full path to override which geckodriver 

95 binary to use for Firefox 47.0.1 and greater, which 

96 defaults to picking up the binary from the system path. 

97 :param options: Instance of ``options.Options``. 

98 :param service_log_path: Where to log information from the driver. 

99 :param firefox_options: Deprecated argument for options 

100 :param service_args: List of args to pass to the driver service 

101 :param desired_capabilities: alias of capabilities. In future 

102 versions of this library, this will replace 'capabilities'. 

103 This will make the signature consistent with RemoteWebDriver. 

104 :param log_path: Deprecated argument for service_log_path 

105 :param keep_alive: Whether to configure remote_connection.RemoteConnection to use 

106 HTTP keep-alive. 

107 """ 

108 if log_path: 

109 warnings.warn('use service_log_path instead of log_path', 

110 DeprecationWarning, stacklevel=2) 

111 service_log_path = log_path 

112 if firefox_options: 

113 warnings.warn('use options instead of firefox_options', 

114 DeprecationWarning, stacklevel=2) 

115 options = firefox_options 

116 self.binary = None 

117 self.profile = None 

118 self.service = None 

119 

120 # If desired capabilities is set, alias it to capabilities. 

121 # If both are set ignore desired capabilities. 

122 if capabilities is None and desired_capabilities: 

123 capabilities = desired_capabilities 

124 

125 if capabilities is None: 

126 capabilities = DesiredCapabilities.FIREFOX.copy() 

127 if options is None: 

128 options = Options() 

129 

130 capabilities = dict(capabilities) 

131 

132 if capabilities.get("binary"): 

133 self.binary = capabilities["binary"] 

134 

135 # options overrides capabilities 

136 if options is not None: 

137 if options.binary is not None: 

138 self.binary = options.binary 

139 if options.profile is not None: 

140 self.profile = options.profile 

141 

142 # firefox_binary and firefox_profile 

143 # override options 

144 if firefox_binary is not None: 

145 if isinstance(firefox_binary, basestring): 

146 firefox_binary = FirefoxBinary(firefox_binary) 

147 self.binary = firefox_binary 

148 options.binary = firefox_binary 

149 if firefox_profile is not None: 

150 if isinstance(firefox_profile, basestring): 

151 firefox_profile = FirefoxProfile(firefox_profile) 

152 self.profile = firefox_profile 

153 options.profile = firefox_profile 

154 

155 # W3C remote 

156 # TODO(ato): Perform conformance negotiation 

157 

158 if capabilities.get("marionette"): 

159 capabilities.pop("marionette") 

160 self.service = Service( 

161 executable_path, 

162 service_args=service_args, 

163 log_path=service_log_path) 

164 self.service.start() 

165 

166 capabilities.update(options.to_capabilities()) 

167 

168 executor = FirefoxRemoteConnection( 

169 remote_server_addr=self.service.service_url) 

170 RemoteWebDriver.__init__( 

171 self, 

172 command_executor=executor, 

173 desired_capabilities=capabilities, 

174 keep_alive=True) 

175 

176 # Selenium remote 

177 else: 

178 if self.binary is None: 

179 self.binary = FirefoxBinary() 

180 if self.profile is None: 

181 self.profile = FirefoxProfile() 

182 

183 # disable native events if globally disabled 

184 self.profile.native_events_enabled = ( 

185 self.NATIVE_EVENTS_ALLOWED and self.profile.native_events_enabled) 

186 

187 if proxy is not None: 

188 proxy.add_to_capabilities(capabilities) 

189 

190 executor = ExtensionConnection("127.0.0.1", self.profile, 

191 self.binary, timeout) 

192 RemoteWebDriver.__init__( 

193 self, 

194 command_executor=executor, 

195 desired_capabilities=capabilities, 

196 keep_alive=keep_alive)
