92class Bucketer(object): 

93 """Collect tests into buckets with similar setup requirements.""" 

94 

95 def __init__(self): 

96 """Initialize the test buckets.""" 

97 # { (frozenset(['users.json']), True): 

98 # [ContextSuite(...), ContextSuite(...)] } 

99 self.buckets = {} 

100 

101 # All the non-FastFixtureTestCase tests we saw, in the order they came 

102 # in: 

103 self.remainder = []

105 def add(self, test): 

106 """Add test into an initialization bucket. 

107 

108 Tests are bucketed according to its set of fixtures and the 

109 value of its exempt_from_fixture_bundling attr. 

110 """ 

111 if is_subclass_at_all(test.context, FastFixtureTestCase): 

112 # We bucket even FFTCs that don't have any fixtures, but it 

113 # shouldn't matter. 

114 key = (frozenset(getattr(test.context, 'fixtures', [])), 

115 getattr(test.context, 

116 'exempt_from_fixture_bundling', 

117 False)) 

118 self.buckets.setdefault(key, []).append(test) 

119 else: 

120 self.remainder.append(test)

144 def _put_transaction_test_cases_last(self, test): 

145 """Reorder test suite so TransactionTestCase-based tests come last. 

146 

147 Django has a weird design decision wherein TransactionTestCase doesn't 

148 clean up after itself. Instead, it resets the DB to a clean state only 

149 at the *beginning* of each test: 

150 https://docs.djangoproject.com/en/dev/topics/testing/?from=olddocs# 

151 django. test.TransactionTestCase. Thus, Django reorders tests so 

152 TransactionTestCases all come last. Here we do the same. 

153 

154 "I think it's historical. We used to have doctests also, adding cleanup 

155 after each unit test wouldn't necessarily clean up after doctests, so 

156 you'd have to clean on entry to a test anyway." was once uttered on 

157 #django-dev. 

158 """ 

159 def filthiness(test): 

160 """Return a score of how messy a test leaves the environment. 

161 

162 Django's TransactionTestCase doesn't clean up the DB on teardown, 

163 but it's hard to guess whether subclasses (other than TestCase) do. 

164 We will assume they don't, unless they have a 

165 ``cleans_up_after_itself`` attr set to True. This is reasonable 

166 because the odd behavior of TransactionTestCase is documented, so 

167 subclasses should by default be assumed to preserve it. 

168 

169 Thus, things will get these comparands (and run in this order): 

170 

171 * 1: TestCase subclasses. These clean up after themselves. 

172 * 1: TransactionTestCase subclasses with 

173 cleans_up_after_itself=True. These include 

174 FastFixtureTestCases. If you're using the 

175 FixtureBundlingPlugin, it will pull the FFTCs out, reorder 

176 them, and run them first of all. 

177 * 2: TransactionTestCase subclasses. These leave a mess. 

178 * 2: Anything else (including doctests, I hope). These don't care 

179 about the mess you left, because they don't hit the DB or, if 

180 they do, are responsible for ensuring that it's clean (as per 

181 https://docs.djangoproject.com/en/dev/topics/testing/?from= 

182 olddocs#writing-doctests) 

183 

184 """ 

185 test_class = test.context 

186 if (is_subclass_at_all(test_class, TestCase) or 

187 (is_subclass_at_all(test_class, TransactionTestCase) and 

188 getattr(test_class, 'cleans_up_after_itself', False))): 

189 return 1 

190 return 2 

191 

192 flattened = [] 

193 process_tests(test, flattened.append) 

194 flattened.sort(key=filthiness) 

195 return ContextSuite(flattened)

197 def _bundle_fixtures(self, test): 

198 """Reorder tests to minimize fixture loading. 

199 

200 I reorder FastFixtureTestCases so ones using identical sets 

201 of fixtures run adjacently. I then put attributes on them 

202 to advise them to not reload the fixtures for each class. 

203 

204 This takes support.mozilla.com's suite from 123s down to 94s. 

205 

206 FastFixtureTestCases are the only ones we care about, because 

207 nobody else, in practice, pays attention to the ``_fb`` advisory 

208 bits. We return those first, then any remaining tests in the 

209 order they were received. 

210 """ 

211 def suite_sorted_by_fixtures(suite): 

212 """Flatten and sort a tree of Suites by fixture. 

213 

214 Add ``_fb_should_setup_fixtures`` and 

215 ``_fb_should_teardown_fixtures`` attrs to each test class to advise 

216 it whether to set up or tear down (respectively) the fixtures. 

217 

218 Return a Suite. 

219 

220 """ 

221 bucketer = Bucketer() 

222 process_tests(suite, bucketer.add) 

223 

224 # Lay the bundles of common-fixture-having test classes end to end 

225 # in a single list so we can make a test suite out of them: 

226 flattened = [] 

227 for (key, fixture_bundle) in bucketer.buckets.items(): 

228 fixtures, is_exempt = key 

229 # Advise first and last test classes in each bundle to set up 

230 # and tear down fixtures and the rest not to: 

231 if fixtures and not is_exempt: 

232 # Ones with fixtures are sure to be classes, which means 

233 # they're sure to be ContextSuites with contexts. 

234 

235 # First class with this set of fixtures sets up: 

236 first = fixture_bundle[0].context 

237 first._fb_should_setup_fixtures = True 

238 

239 # Set all classes' 1..n should_setup to False: 

240 for cls in fixture_bundle[1:]: 

241 cls.context._fb_should_setup_fixtures = False 

242 

243 # Last class tears down: 

244 last = fixture_bundle[-1].context 

245 last._fb_should_teardown_fixtures = True 

246 

247 # Set all classes' 0..(n-1) should_teardown to False: 

248 for cls in fixture_bundle[:-1]: 

249 cls.context._fb_should_teardown_fixtures = False 

250 

251 flattened.extend(fixture_bundle) 

252 flattened.extend(bucketer.remainder) 

253 

254 return ContextSuite(flattened) 

255 

256 return suite_sorted_by_fixtures(test)

258 def prepareTest(self, test): 

259 """Reorder the tests.""" 

260 test = self._put_transaction_test_cases_last(test) 

261 if self.should_bundle: 

262 test = self._bundle_fixtures(test) 

263 return test
