63 def begin(self): 

64 """ 

65 Called before any tests are run.

67 The only trick here is that we have to mangle the order of

140 def _attach_attributes(self, fullname, obj): 

141 """ 

142 Attach attributes matching 'fullname' to the object 'obj'. 

143 """ 

144 attribs = self.curtains.get(fullname, []) 

145 log.info('_attach_attributes: %s, %s' % (fullname, attribs,))
